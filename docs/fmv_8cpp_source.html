43void FMVFilter::set_parameters(float l, float m, float t)
44{
45 const float R1 = 220e3;
46 const float R2 = 1e6;
47 const float R3 = 22e3;
48 const float R4 = 33e3;
49
50 const float C1 = 470e-12;
51 const float C2 = 22e-9;
52 const float C3 = 22e-9;
53
54 float b1 = t * C1 * R1 + m * C3 * R3 + l * (C1 * R2 + C2 * R2) +
55 (C1 * R3 + C2 * R3);
56
57 float b2 = t * (C1 * C2 * R1 * R4 + C1 * C3 * R1 * R4 )
58 - m * m * (C1 * C3 * R3 * R3 + C2 * C3 * R3 * R3 )
59 + m * (C1 * C3 * R1 * R3 + C1 * C3 * R3 * R3 + C2 * C3 * R3 * R3)
60 + l * (C1 * C2 * R1 * R2 + C1 * C2 * R2 * R4 + C1 * C3 * R2 * R4)
61 + l * m * (C1 * C3 * R2 * R3 + C2 * C3 * R2 * R3)
62 + (C1 * C2 * R1 * R3 + C1 * C2 * R3 * R4 + C1 * C3 * R3 * R4);
63
64 float b3 = l*m*(C1 * C2 * C3 * R1 * R2 * R3 + C1 * C2 * C3 * R2 * R3 * R4 )
65 - m * m * (C1 * C2 * C3 * R1 * R3 * R3 + C1 * C2 * C3 * R3 * R3 * R4 )
66 + m * (C1 * C2 * C3 * R1 * R3 * R3 + C1 * C2 * C3 * R3 * R3 * R4 )
67 + t * C1 * C2 * C3 * R1 * R3 * R4 - t * m * C1 * C2 * C3 * R1 * R3 * R4
68 + t * l * C1 * C2 * C3 * R1 * R2 * R4;
69
70 float a0 = 1;
71
72 float a1 = (C1 * R1 + C1 * R3 + C2 * R3 + C2 * R4 + C3 * R4 )
73 + m * C3 * R3 + l * (C1 * R2 + C2 * R2);
74
75 float a2 = m * (C1 * C3 * R1 * R3 - C2 * C3 * R3 * R4 + C1 * C3 * R3 * R3
76 + C2 * C3 * R3 * R3 ) + l * m * (C1 * C3 * R2 * R3 + C2 * C3 * R2 * R3 )
77 - m*m * (C1 * C3 * R3 * R3 + C2 * C3 * R3 * R3 ) + l*(C1 * C2 * R2 * R4
78 + C1 * C2 * R1 * R2 + C1 * C3 * R2 * R4 + C2 * C3 * R2 * R4 )
79 + (C1 * C2 * R1 * R4 + C1 * C3 * R1 * R4 + C1 * C2 * R3 * R4
80 + C1 * C2 * R1 * R3 + C1 * C3 * R3 * R4 + C2 * C3 * R3 * R4 );
81
82 float a3 = l*m*(C1 * C2 * C3 * R1 * R2 * R3 + C1 * C2 * C3 * R2 * R3 * R4 )
83 - m * m * (C1 * C2 * C3 * R1 * R3 * R3 + C1 * C2 * C3 * R3 * R3 * R4)
84 + m * (C1 * C2 * C3 * R3 * R3 * R4 + C1 * C2 * C3 * R1 * R3 * R3
85 - C1 * C2 * C3 * R1 * R3 * R4 )
86 + l * C1 * C2 * C3 * R1 * R2 * R4 + C1 * C2 * C3 * R1 * R3 * R4;
87
88 //bilinear transform
89 float c = 2*samplerate;
90 float c2 = c*c;
91 float c3 = c*c*c;
92
93 float B0 = -b1 * c- b2 * c2 - b3 * c3;
94 float B1 = -b1 * c+ b2 * c2 + 3 * b3 * c3;
95 float B2 = b1 * c+ b2 * c2 - 3 * b3 * c3;
96 float B3 = b1 * c- b2 * c2 + b3 * c3;
97
98 float A0 = -a0 - a1 * c- a2 * c2 - a3 * c3;
99 float A1 = -3 * a0 - a1 * c+ a2 * c2 + 3 * a3 * c3;
100 float A2 = -3 * a0 + a1 * c+ a2 * c2 - 3 * a3 * c3;
101 float A3 = -a0 + a1 * c- a2 * c2 + a3 * c3;
102
103 filter.set_coefficients(std::array<float, 8>{B0, B1, B2, B3, A0, A1, A2, A3});
104}
