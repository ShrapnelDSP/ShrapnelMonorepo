67 void work(void)
68 {
69 /* TODO should not leave these uninitialised */
70 char *message_type;
71 cJSON *type;
72
73 int ret = queue->receive(&message, portMAX_DELAY);
74 if(ret != pdTRUE)
75 {
76 ESP_LOGE(TAG, "Queue failed to receive");
77 return;
78 }
79
80#if !defined(TESTING)
81 ESP_LOGI(TAG, "%s stack %d", __FUNCTION__, uxTaskGetStackHighWaterMark(NULL));
82#endif
83 size_t message_size = sizeof(message.json);
84 assert(message_size <= INT_MAX);
85 (void)message_size;
86
87 ESP_LOGI(TAG, "received websocket message: %.*s",
88 static_cast<int>(sizeof(message.json)),
89 message.json);
90
91 json = cJSON_ParseWithLength(message.json, sizeof(message.json));
92 if(json == NULL)
93 {
94 ESP_LOGE(TAG, "json parsing failed");
95 goto done;
96 }
97
98 type = cJSON_GetObjectItemCaseSensitive(json, "messageType");
99 if(cJSON_IsString(type) && (type->valuestring != NULL))
100 {
101 message_type = type->valuestring;
102 }
103 else
104 {
105 ESP_LOGE(TAG, "error parsing messageType from json");
106 goto done;
107 }
108
109 if(0 == strcmp(message_type, "initialiseParameters"))
110 {
111 initialise_parameters();
112 }
113 else if(0 == strcmp(message_type, "parameterUpdate"))
114 {
115 parameter_update();
116 }
117 else
118 {
119 ESP_LOGE(TAG, "unknown message type (%s)", message_type);
120 }
121
122done:
123 cJSON_Delete(json);
124 }
125
126 private:
127 void parameter_update(void)
128 {
129 assert(json);
130
131 char *parsed_id;
132 cJSON *id = cJSON_GetObjectItemCaseSensitive(json, "id");
133 if(cJSON_IsString(id) && (id->valuestring != NULL))
134 {
135 parsed_id = id->valuestring;
136 }
137 else
138 {
139 ESP_LOGE(TAG, "error parsing id from json");
140 return;
141 }
142
143 float parsed_value;
144 cJSON *value = cJSON_GetObjectItemCaseSensitive(json, "value");
145 if(cJSON_IsNumber(value))
146 {
147 parsed_value = value->valuedouble;
148 }
149 else
150 {
151 ESP_LOGE(TAG, "error parsing value from json");
152 return;
153 }
154
155 int rc = param->update(parsed_id, parsed_value);
156 if(rc != 0)
157 {
158 ESP_LOGE(TAG, "Failed to update parameter (%s) with value %f", parsed_id, parsed_value);
159 }
160
161 event.send(message.json, message.fd);
162 }
163
164 void initialise_parameters(void)
165 {
166 assert(json);
167
168 Message output;
169
170 for(const auto& [key, value] : *param)
171 {
172 float tmp_f = value->get();
173
174 snprintf(output.json, sizeof(output.json),
175 "{\"id\":\"%s\",\"value\":%g}",
176 key.c_str(),
177 tmp_f);
178
179 event.send(output.json, -1);
180 }
181 }

187 cJSON *json;
188 Message message;
