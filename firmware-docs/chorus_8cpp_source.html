27float triangle(float phase)
28{
29 if(phase < (float)M_PI)
30 {
31 return 1 - 2 * phase / (float)M_PI;
32 }
33 else
34 {
35 return - 3 + 2 * phase / (float)M_PI;
36 }
37}

63void Chorus::process(float *samples, std::size_t sample_count)
64{
65 for(std::size_t i = 0; i < sample_count; i++)
66 {
67 float lfo = 0.5f * triangle(phase);
68 phase += modulation_rate / sample_rate * 2 * (float)M_PI;
69
70 if(phase > 2 * (float)M_PI)
71 {
72 phase -= 2 * (float)M_PI;
73 }
74
75#if 1
76 float delay = MAX_DELAY_MS / 1000 * sample_rate *
77 (0.5f + (modulation_depth * lfo));
78
79 dspal_delayline_set_delay(delayline, delay);
80
81 dspal_delayline_push_sample(delayline, samples[i]);
82 samples[i] = samples[i] +
83 (modulation_mix * dspal_delayline_pop_sample(delayline));
84#else
85 samples[i] = lfo;
86#endif
87 }
88}
