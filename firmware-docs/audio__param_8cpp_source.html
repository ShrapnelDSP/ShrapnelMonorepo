38void AudioParameterFloat::update(float a_value)
39{
40 if(a_value > 1)
41 {
42 return;
43 }
44
45 if(a_value < 0)
46 {
47 return;
48 }
49
50 auto range = maximum - minimum;
51 value = minimum + a_value * range;
52}
53
54float AudioParameterFloat::get(void)
55{
56 auto range = maximum - minimum;
57 return (value - minimum) / range;
58}

60std::atomic<float> *AudioParameterFloat::get_raw_parameter(void)
61{
62 return &value;
63}

83int AudioParameters::update(const id_t &param, float value)
84{
85 (void) param;
86 (void) value;
87
88 auto element = parameters.find(param);
89
90 if(element == parameters.end())
91 {
92 return -1;
93 }
94
95 parameters[param]->update(value);
96
97 ESP_LOGI(TAG, "Updated %.*s to %f", param.length(), param.data(), value);
98
99 return 0;
100}
