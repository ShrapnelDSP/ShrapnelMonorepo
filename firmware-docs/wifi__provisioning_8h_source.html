56static void provisioning_event_handler(void* arg, esp_event_base_t event_base,
57 int32_t event_id, void* event_data)
58{
59 assert(arg != nullptr);
60 assert(event_base == WIFI_PROV_EVENT);
61
62 auto queue = static_cast<Queue<Event, 1> *>(arg);
63
64 if (event_base == WIFI_PROV_EVENT) {
65 switch (event_id) {
66 case WIFI_PROV_INIT:
67 ESP_LOGI(TAG, "Provisioning initialised");
68 break;
69 case WIFI_PROV_START:
70 ESP_LOGI(TAG, "Provisioning started");
71 break;
72 case WIFI_PROV_CRED_RECV: {
73 wifi_sta_config_t *wifi_sta_cfg = (wifi_sta_config_t *)event_data;
74 ESP_LOGI(TAG, "Received Wi-Fi credentials");
75 ESP_LOGI(TAG, "SSID:");
76 ESP_LOG_BUFFER_HEXDUMP(
77 TAG,
78 wifi_sta_cfg->ssid,
79 sizeof(wifi_sta_cfg->ssid),
80 ESP_LOG_INFO);
81 ESP_LOGI(TAG, "Password:");
82 ESP_LOG_BUFFER_HEXDUMP( TAG,
83 wifi_sta_cfg->password,
84 sizeof(wifi_sta_cfg->password),
85 ESP_LOG_INFO);
86 break;
87 }
88 case WIFI_PROV_CRED_FAIL: {
89 auto reason = (wifi_prov_sta_fail_reason_t *)event_data;
90 ESP_LOGE(TAG, "Provisioning failed!\n\tReason : %s",
91 (*reason == WIFI_PROV_STA_AUTH_ERROR) ?
92 "Wi-Fi station authentication failed" : "Wi-Fi access-point not found");
93
94 Event event{.type = EVENT_PROVISIONING_FAILED};
95 int rc = queue->send(&event, 100 / portTICK_PERIOD_MS);
96 if(rc != true)
97 {
98 ESP_LOGE(TAG, "failed to send to queue");
99 }
100 break;
101 }
102 case WIFI_PROV_CRED_SUCCESS: {
103 ESP_LOGI(TAG, "Provisioning successful");
104
105 Event event{.type = EVENT_PROVISIONING_SUCCESSFUL};
106 int rc = queue->send(&event, 100 / portTICK_PERIOD_MS);
107 if(rc != true)
108 {
109 ESP_LOGE(TAG, "failed to send to queue");
110 }
111 break;
112 }
113 case WIFI_PROV_END:
114 ESP_LOGI(TAG, "Provisioning end");
115 break;
116 default:
117 ESP_LOGW(TAG, "Unhandled provisioning event: %d", event_id);
118 break;
119 }
120 }
121}

123static void wifi_event_handler(void* arg, esp_event_base_t event_base,
124 int32_t event_id, void* event_data)
125{
126 (void) arg;
127
128 if (event_base == WIFI_EVENT && event_id == WIFI_EVENT_SCAN_DONE) {
129 ESP_LOGI(TAG, "WIFI_EVENT_SCAN_DONE");
130 } else if (event_base == WIFI_EVENT && event_id == WIFI_EVENT_STA_START) {
131 ESP_LOGI(TAG, "WIFI_EVENT_STA_START");
132 } else if (event_base == WIFI_EVENT && event_id == WIFI_EVENT_STA_CONNECTED) {
133 ESP_LOGI(TAG, "WIFI_EVENT_STA_CONNECTED");
134 } else if (event_base == WIFI_EVENT && event_id == WIFI_EVENT_STA_DISCONNECTED) {
135 ESP_LOGE(TAG, "WIFI_EVENT_STA_DISCONNECTED");
136 } else if (event_base == IP_EVENT && event_id == IP_EVENT_STA_GOT_IP) {
137 ip_event_got_ip_t* event = (ip_event_got_ip_t*) event_data;
138 ESP_LOGI(TAG, "Connected with IP Address:" IPSTR, IP2STR(&event->ip_info.ip));
139 } else if (event_base == WIFI_EVENT && event_id == WIFI_EVENT_AP_START) {
140 ESP_LOGI(TAG, "WIFI_EVENT_AP_START");
141 } else if (event_base == WIFI_EVENT && event_id == WIFI_EVENT_AP_STOP) {
142 ESP_LOGI(TAG, "WIFI_EVENT_AP_STOP");
143 } else if (event_base == WIFI_EVENT && event_id == WIFI_EVENT_AP_STACONNECTED) {
144 ESP_LOGI(TAG, "WIFI_EVENT_AP_STACONNECTED");
145 } else if (event_base == WIFI_EVENT && event_id == WIFI_EVENT_AP_STADISCONNECTED) {
146 ESP_LOGE(TAG, "WIFI_EVENT_AP_STADISCONNECTED");
147 } else {
148 ESP_LOGW(TAG, "Unhandled wifi event: %s %d", event_base, event_id);
149 }
150}

152static void get_device_service_name(char *service_name, size_t max)
153{
154 uint8_t eth_mac[6];
155 const char *ssid_prefix = "shrapnel_";
156 esp_wifi_get_mac(WIFI_IF_STA, eth_mac);
157 snprintf(service_name, max, "%s%02X%02X%02X",
158 ssid_prefix, eth_mac[3], eth_mac[4], eth_mac[5]);
159}

195static void start(void)
196{
197 ESP_LOGI(TAG, "Starting provisioning");
198
199 char service_name[16];
200 get_device_service_name(service_name, sizeof(service_name));
201
202 wifi_prov_security_t security = WIFI_PROV_SECURITY_1;
203 constexpr std::string_view proof_of_possession{"abcd1234"};
204 wifi_prov_security1_params_t params{
205 .data{reinterpret_cast<const uint8_t *>(proof_of_possession.data())},
206 .len{proof_of_possession.size()},
207 };
208 const char *service_key = NULL;
209
210 ESP_ERROR_CHECK(wifi_prov_mgr_start_provisioning(security, &params, service_name, service_key));
211}

213void wait_for_provisioning(void)
214{
215 start();
216
217 while(1)
218 {
219 Event event;
220 int rc = queue.receive(&event, portMAX_DELAY);
221 assert(rc == true);
222
223 if(event.type == EVENT_PROVISIONING_SUCCESSFUL)
224 {
225 break;
226 }
227 else if(event.type == EVENT_PROVISIONING_FAILED)
228 {
229 wifi_prov_mgr_stop_provisioning();
230 wifi_prov_mgr_wait();
231 start();
232 }
233 else
234 {
235 ESP_LOGE(TAG, "unhandled event %d", event.type);
236 }
237 }
238
239 ESP_LOGI(TAG, "Provisioning finished");
240}
