29int EspStorage::save(const char *key, std::span<uint8_t> data)
30{
31 nvs_handle_t nvs_handle;
32 esp_err_t err;
33 int rc = -1;
34
35 ESP_ERROR_CHECK_WITHOUT_ABORT(
36 err = nvs_open(STORAGE_NAMESPACE, NVS_READWRITE, &nvs_handle));
37 if(err != ESP_OK)
38 goto out;
39
40 ESP_ERROR_CHECK_WITHOUT_ABORT(
41 err = nvs_set_blob(nvs_handle, key, data.data(), data.size()));
42 if(err != ESP_OK)
43 goto out;
44
45 ESP_ERROR_CHECK_WITHOUT_ABORT(err = nvs_commit(nvs_handle));
46 if(err != ESP_OK)
47 goto out;
48
49 rc = 0;
50
51out:
52 nvs_close(nvs_handle);
53 return rc;
54}

146int EspStorage::load(const char *key, std::span<uint8_t> &buffer)
147{
148 nvs_handle_t nvs_handle;
149 esp_err_t err;
150 std::size_t required_size = 0;
151 int rc = -1;
152
153 ESP_ERROR_CHECK_WITHOUT_ABORT(
154 err = nvs_open(STORAGE_NAMESPACE, NVS_READONLY, &nvs_handle));
155 if(err != ESP_OK)
156 goto out;
157
158 // query the required size
159 ESP_ERROR_CHECK_WITHOUT_ABORT(
160 err = nvs_get_blob(nvs_handle, key, nullptr, &required_size));
161 if(err != ESP_OK)
162 goto out;
163
164 if(required_size > buffer.size())
165 {
166 ESP_LOGE(TAG,
167 "Not enough space in the output buffer. Need %zu, got %zu",
168 required_size,
169 buffer.size());
170 goto out;
171 }
172
173 ESP_ERROR_CHECK_WITHOUT_ABORT(
174 err = nvs_get_blob(nvs_handle, key, buffer.data(), &required_size));
175 if(err != ESP_OK)
176 goto out;
177
178 buffer = buffer.subspan(0, required_size);
179 rc = 0;
180out:
181 nvs_close(nvs_handle);
182 return rc;
183}

185int EspStorage::load(const char *key, etl::istring &data)
186{
187 nvs_handle_t nvs_handle;
188 esp_err_t err;
189 std::size_t required_size = 0;
190 int rc = -1;
191
192 etl::string<15> truncated_key{key};
193
194 ESP_ERROR_CHECK_WITHOUT_ABORT(
195 err = nvs_open(STORAGE_NAMESPACE, NVS_READONLY, &nvs_handle));
196 if(err != ESP_OK)
197 goto out;
198
199 // query the required size
200 ESP_ERROR_CHECK_WITHOUT_ABORT(
201 err = nvs_get_blob(
202 nvs_handle, truncated_key.data(), nullptr, &required_size));
203 if(err != ESP_OK)
204 goto out;
205
206 data.initialize_free_space();
207 data.uninitialized_resize(required_size);
208
209 if(data.size() != required_size)
210 {
211 ESP_LOGE(TAG,
212 "string is too small, need %zu, got %zu",
213 required_size,
214 data.size());
215 return -1;
216 }
217
218 ESP_ERROR_CHECK_WITHOUT_ABORT(
219 err = nvs_get_blob(
220 nvs_handle, truncated_key.data(), data.data(), &required_size));
221 if(err != ESP_OK)
222 goto out;
223
224 rc = 0;
225out:
226 nvs_close(nvs_handle);
227 return rc;
228}
