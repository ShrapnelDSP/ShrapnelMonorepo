38int EspStorage::save(const char *key, std::span<uint8_t> data)
39{
40 esp_err_t err;
41
42 ESP_ERROR_CHECK_WITHOUT_ABORT(
43 err = nvs_set_blob(nvs_handle, key, data.data(), data.size()));
44 if(err != ESP_OK)
45 return -1;
46
47 ESP_ERROR_CHECK_WITHOUT_ABORT(err = nvs_commit(nvs_handle));
48 if(err != ESP_OK)
49 return -1;
50
51 return 0;
52}

111int EspStorage::load(const char *key, std::span<uint8_t> &buffer)
112{
113 esp_err_t err;
114 std::size_t required_size = 0;
115
116 // query the required size
117 ESP_ERROR_CHECK_WITHOUT_ABORT(
118 err = nvs_get_blob(nvs_handle, key, nullptr, &required_size));
119 if(err != ESP_OK)
120 return -1;
121
122 if(required_size > buffer.size())
123 {
124 ESP_LOGE(TAG,
125 "Not enough space in the output buffer. Need %zu, got %zu",
126 required_size,
127 buffer.size());
128 return -1;
129 }
130
131 ESP_ERROR_CHECK_WITHOUT_ABORT(
132 err = nvs_get_blob(nvs_handle, key, buffer.data(), &required_size));
133 if(err != ESP_OK)
134 return -1;
135
136 buffer = buffer.subspan(0, required_size);
137 return 0;
138}

140int EspStorage::load(const char *key, etl::istring &data)
141{
142 esp_err_t err;
143 std::size_t required_size = 0;
144
145 etl::string<15> truncated_key{key};
146
147 // query the required size
148 ESP_ERROR_CHECK_WITHOUT_ABORT(
149 err = nvs_get_blob(
150 nvs_handle, truncated_key.data(), nullptr, &required_size));
151 if(err != ESP_OK)
152 return -1;
153
154 data.initialize_free_space();
155 data.uninitialized_resize(required_size);
156
157 if(data.size() != required_size)
158 {
159 ESP_LOGE(TAG,
160 "string is too small, need %zu, got %zu",
161 required_size,
162 data.size());
163 return -1;
164 }
165
166 ESP_ERROR_CHECK_WITHOUT_ABORT(
167 err = nvs_get_blob(
168 nvs_handle, truncated_key.data(), data.data(), &required_size));
169 if(err != ESP_OK)
170 return -1;
171
172 return 0;
173}
