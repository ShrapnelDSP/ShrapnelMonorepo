29int EspStorage::save(const char *key, etl::string_view data) {
30 nvs_handle_t nvs_handle;
31 esp_err_t err;
32 int rc = -1;
33
34 etl::string<15> truncated_key{key};
35
36 ESP_ERROR_CHECK_WITHOUT_ABORT(err = nvs_open(STORAGE_NAMESPACE, NVS_READWRITE, &nvs_handle));
37 if (err != ESP_OK) goto out;
38
39 ESP_ERROR_CHECK_WITHOUT_ABORT(err = nvs_set_blob(nvs_handle, truncated_key.data(), data.data(), data.size()));
40 if (err != ESP_OK) goto out;
41
42 ESP_ERROR_CHECK_WITHOUT_ABORT(err = nvs_commit(nvs_handle));
43 if (err != ESP_OK) goto out;
44
45 rc = 0;
46
47out:
48 nvs_close(nvs_handle);
49 return rc;
50}

52int EspStorage::load(const char *key, etl::istring &data) {
53 nvs_handle_t nvs_handle;
54 esp_err_t err;
55 std::size_t required_size = 0;
56 int rc = -1;
57
58 etl::string<15> truncated_key{key};
59
60 ESP_ERROR_CHECK_WITHOUT_ABORT(err = nvs_open(STORAGE_NAMESPACE, NVS_READONLY, &nvs_handle));
61 if (err != ESP_OK) goto out;
62
63 // query the required size
64 ESP_ERROR_CHECK_WITHOUT_ABORT(err = nvs_get_blob(nvs_handle, truncated_key.data(), nullptr, &required_size));
65 if (err != ESP_OK) goto out;
66
67 data.initialize_free_space();
68 data.uninitialized_resize(required_size);
69
70 if(data.size() != required_size)
71 {
72 ESP_LOGE(TAG, "string is too small, need %zu, got %zu", required_size, data.size());
73 return -1;
74 }
75
76 ESP_ERROR_CHECK_WITHOUT_ABORT(err = nvs_get_blob(nvs_handle, truncated_key.data(), data.data(), &required_size));
77 if (err != ESP_OK) goto out;
78
79 rc = 0;
80out:
81 nvs_close(nvs_handle);
82 return rc;
83}
