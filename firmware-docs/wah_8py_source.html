37def design_filter(wah):
38 # Identical to implementation in [2] except the gain tweaked to better
39 # match measurements at [1], then further tweaked for constant loudness
40 # across the sweep.
41 # This is only used for reference plot, and not actually implemented in
42 # firmware.
43 g = 0.1*math.e**(1.75*wah)
44 fr = 450*2**(2.3*wah)
45 Q_value = 2**(2*(1-wah)+1)
46
47 frn = fr/fs;
48 R = 1 - math.pi*frn/Q_value; # pole radius
49 theta = 2*math.pi*frn; # pole angle
50 a1 = -2.0*R*math.cos(theta); # biquad coeff
51 a2 = R*R; # biquad coeff
52
53 return ([g, -g, 0], [1, a1, a2])

57def design_filter2(wah, vocal):
58 # Reimagined based on digital prototype filter with more preditcable
59 # parameters, and added vocal control that reduces the Q.
60 # Parameters tweaked for more consistent loudness across the sweep.
61 # This is the version implemented in the firmware.
62
63 # parameters to normalise loudness. Larger values mean more
64 # correlation. Negative values mean the gain decreases as the parameter
65 # increases
66 g_factor_wah = -1.2
67 g_factor_vocal = -1
68
69 g = 14 * math.e**(g_factor_wah * wah + g_factor_vocal * vocal)
70 f = 450*2**(2.3*wah)
71 qFactor = 2**(2*(1-wah)+1 - 2 * vocal)
72
73 # esp-dsp dsps_biquad_gen_bpf0db_f32
74 w0 = 2 * math.pi * f / fs;
75 c = math.cos(w0);
76 s = math.sin(w0);
77 alpha = s / (2 * qFactor);
78
79 b0 = alpha;
80 b1 = 0;
81 b2 = -b0;
82 a0 = 1 + alpha;
83 a1 = -2 * c;
84 a2 = 1 - alpha;
85
86 return ([ g * b0 / a0, g * b1 / a0, g * b2 / a0], [1, a1 / a0, a2 / a0])
