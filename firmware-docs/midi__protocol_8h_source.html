75 friend etl::string_stream& operator<<(etl::string_stream& out, const Message& message) {
76 out << "{ channel " << message.channel << " ";
77
78 auto print_parameters = [&](auto &param) {
79 using T = std::decay_t<decltype(param)>;
80
81 if constexpr (std::is_same_v<T, NoteOn>) {
82 out << "note on " << +param.note << " " << +param.velocity;
83 } else if constexpr (std::is_same_v<T, NoteOff>) {
84 out << "note off " << +param.note << " " << +param.velocity;
85 } else if constexpr (std::is_same_v<T, ControlChange>) {
86 out << "control change " << +param.control << " " << +param.value;
87 } else if constexpr (std::is_same_v<T, ProgramChange>) {
88 out << "program change " << +param.number;
89 } else {
90 // TODO is there any way to turn this into a compiler error?
91 out << "unknown";
92 }
93 };
94
95 std::visit(print_parameters, message.parameters);
96
97 out << " }";
98 return out;
99 }
