87void process_samples(int32_t *buf, size_t buf_len)
88{
89 assert(g_profiling_gpio != -1 && "I2S task has not been initialised");
90
91 gpio_set_level(g_profiling_gpio, 1);
92 profiling_start();
93 /* Copy samples, skipping every second one. These correspond to the left
94 * channel of the ADC and DAC, which are not used here
95 * XXX not sure why the right channel is not at index 0 */
96 for(size_t i = 1; i < buf_len; i+=2)
97 {
98 fbuf[i/2] = buf[i]/(float)INT32_MAX;
99 }
100
101 gate_set_threshold(*gate_threshold, *gate_hysteresis);
102 gate_set_attack(*gate_attack);
103 gate_set_hold(*gate_hold);
104 gate_set_release(*gate_release);
105
106 profiling_mark_stage(0);
107
108 gate_analyse(fbuf, buf_len/2);
109 profiling_mark_stage(1);
110
111 valvestate->set_gain(*amp_gain, *amp_channel);
112 valvestate->set_fmv(*bass, *middle, *treble);
113 valvestate->set_contour(*contour);
114 valvestate->set_volume(decibel_to_ratio(*volume));
115 profiling_mark_stage(2);
116
117 valvestate->process(fbuf, buf_len/2);
118 profiling_mark_stage(3);
119
120 if(*gate_bypass < 0.5f)
121 {
122 gate_process(fbuf, buf_len/2);
123 }
124 profiling_mark_stage(4);
125
126 /* speaker IR */
127 speaker->process(fbuf);
128 profiling_mark_stage(17);
129
130 chorus->set_modulation_rate_hz(*chorus_rate);
131 chorus->set_modulation_depth(*chorus_depth);
132 chorus->set_modulation_mix(*chorus_mix);
133 profiling_mark_stage(18);
134
135 if(*chorus_bypass < 0.5f)
136 {
137 chorus->process(fbuf, buf_len/2);
138 }
139 profiling_mark_stage(19);
140
141 for(size_t i = 1; i < buf_len; i+=2)
142 {
143 if(fbuf[i/2] > 1.f || fbuf[i/2] < -1.f)
144 {
145 shrapnel::events::output_clipped.clear();
146 shrapnel::events::output_clipped.notify_all();
147 }
148
149 if(fbuf[i/2] != fbuf[i/2])
150 {
151 ESP_LOGE(TAG, "Not a number at index %d", i);
152 assert(0);
153 }
154
155 buf[i] = float_to_int32(fbuf[i/2]);
156 }
157 profiling_mark_stage(20);
158
159 profiling_stop();
160 gpio_set_level(g_profiling_gpio, 0);
161
162 ESP_LOGD(TAG, "stack: %d", uxTaskGetStackHighWaterMark(NULL));
163}

165esp_err_t process_init(shrapnel::parameters::AudioParametersBase *audio_params)
166{
167 ESP_ERROR_CHECK(gate_init());
168 gate_set_sample_rate(SAMPLE_RATE);
169 gate_set_threshold(-60, 1);
170 gate_set_attack(10);
171 gate_set_hold(50);
172 gate_set_release(50);
173 gate_set_buffer_size(DMA_BUF_SIZE);
174
175 chorus = new shrapnel::effect::Chorus();
176 assert(chorus);
177 chorus->set_sample_rate(SAMPLE_RATE);
178
179 valvestate = new shrapnel::effect::valvestate::Valvestate();
180 assert(valvestate);
181 valvestate->prepare(SAMPLE_RATE);
182
183 amp_gain = audio_params->get_raw_parameter("ampGain");
184 assert(amp_gain);
185 amp_channel = audio_params->get_raw_parameter("ampChannel");
186 assert(amp_channel);
187 bass = audio_params->get_raw_parameter("bass");
188 assert(bass);
189 middle = audio_params->get_raw_parameter("middle");
190 assert(middle);
191 treble = audio_params->get_raw_parameter("treble");
192 assert(treble);
193 contour = audio_params->get_raw_parameter("contour");
194 assert(contour);
195 volume = audio_params->get_raw_parameter("volume");
196 assert(volume);
197
198 gate_threshold = audio_params->get_raw_parameter("noiseGateThreshold");
199 assert(gate_threshold);
200 gate_hysteresis = audio_params->get_raw_parameter("noiseGateHysteresis");
201 assert(gate_hysteresis);
202 gate_attack = audio_params->get_raw_parameter("noiseGateAttack");
203 assert(gate_attack);
204 gate_hold = audio_params->get_raw_parameter("noiseGateHold");
205 assert(gate_hold);
206 gate_release = audio_params->get_raw_parameter("noiseGateRelease");
207 assert(gate_release);
208 gate_bypass = audio_params->get_raw_parameter("noiseGateBypass");
209 assert(gate_bypass);
210
211 chorus_rate = audio_params->get_raw_parameter("chorusRate");
212 assert(chorus_rate);
213 chorus_depth = audio_params->get_raw_parameter("chorusDepth");
214 assert(chorus_depth);
215 chorus_mix = audio_params->get_raw_parameter("chorusMix");
216 assert(chorus_mix);
217 chorus_bypass = audio_params->get_raw_parameter("chorusBypass");
218 assert(chorus_bypass);
219
220 ESP_LOGI(TAG, "Initialised FIR filter with length %d", fir_coeff.size());
221
222 auto speaker_convolution = std::make_unique<shrapnel::dsp::FastConvolution<1024, 512>>(fir_coeff);
223 assert(speaker_convolution.get());
224
225 speaker = new shrapnel::dsp::FastFir<
226 DMA_BUF_SIZE,
227 1024,
228 shrapnel::dsp::FastConvolution<1024, 512>>(std::move(speaker_convolution));
229 assert(speaker);
230
231 return ESP_OK;
232}
