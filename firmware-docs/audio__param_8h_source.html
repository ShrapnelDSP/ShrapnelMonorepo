73template <const std::size_t MAX_PARAMETERS, const std::size_t MAX_OBSERVERS>
74class AudioParameters final
75 : public etl::observable<ParameterObserver, MAX_OBSERVERS>,
76 public AudioParametersBase
77{
78public:
79 using MapType =
80 etl::map<id_t, std::unique_ptr<AudioParameterFloat>, MAX_PARAMETERS>;
81
89 [[nodiscard]] int update(const id_t &param, float value)
90 {
91 auto element = parameters.find(param);
92
93 if(element == parameters.end())
94 {
95 return -1;
96 }
97
98 parameters[param]->update(value);
99 this->notify_observers(std::move(std::pair{param, value}));
100
101 ESP_LOGI(TAG,
102 "Updated %.*s to %f",
103 (int)param.length(),
104 param.data(),
105 value);
106
107 return 0;
108 }
109
111 float get(const id_t &param)
112 {
113 auto element = parameters.find(param);
114
115 if(element == parameters.end())
116 {
117 return 0;
118 }
119
120 return parameters[param]->get();
121 }
122
127 std::atomic<float> *get_raw_parameter(const id_t &param) override
128 {
129 auto element = parameters.find(param);
130
131 if(element == parameters.end())
132 {
133 return nullptr;
134 }
135
136 return parameters[param]->get_raw_parameter();
137 }
138
139 [[nodiscard]] int create_and_add_parameter(const id_t &name,
140 float minimum,
141 float maximum,
142 float default_value)
143 {
144 if(parameters.full())
145 {
146 return -1;
147 }
148
149 parameters[name] = std::make_unique<AudioParameterFloat>(
150 name, minimum, maximum, default_value);
151
152 ESP_LOGI(TAG, "%zu parameters are registered", parameters.size());
153 return 0;
154 }
155
156 MapType::iterator begin() { return parameters.begin(); }
157
158 MapType::iterator end() { return parameters.end(); }
159
160private:
161 static constexpr const char *TAG = "audio_param";
162 MapType parameters;
163};
