29inline etl::string<15> id_to_key(id_t id)
30{
31 char hex[9];
32 int rc = snprintf(hex, sizeof hex, "%08" PRIx32, id);
33 assert(rc == 8);
34 return {hex, 9};
35}
36
37inline id_t key_to_id(const etl::string<15> &key)
38{
39 id_t id;
40 int rc = sscanf(key.c_str(), "%08" PRIx32, &id);
41 assert(rc == 1);
42 return id;
43}

53template <size_t READ_BUFFER_SIZE>
54class EspCrud final : public persistence::Crud<std::span<uint8_t>>
55{
56public:
57 EspCrud(const char *a_part_name, const char *a_namespace_name)
58 : part_name{a_part_name},
59 namespace_name{a_namespace_name}
60 {
61 ESP_ERROR_CHECK(nvs_open(namespace_name, NVS_READWRITE, &nvs_handle));
62
63 ESP_LOGI(TAG,
64 "Started EspCrud for partition %s namespace %s with handle "
65 "%" PRIu32,
66 part_name,
67 namespace_name,
68 nvs_handle);
69 }

71 ~EspCrud() final { nvs_close(nvs_handle); }

73 [[nodiscard]] int create(const std::span<uint8_t> &data,
74 uint32_t &id_out) override
75 {
76 esp_err_t err;
77
78 // start at 0 in case last ID has not been saved to NVS
79 uint32_t id = 0;
80
81 ESP_ERROR_CHECK_WITHOUT_ABORT(
82 err = nvs_get_u32(nvs_handle, last_id_name, &id));
83 if(err != ESP_OK && err != ESP_ERR_NVS_NOT_FOUND)
84 return -1;
85
86 id++;
87 id_out = id;
88
89 ESP_ERROR_CHECK_WITHOUT_ABORT(
90 err = nvs_set_blob(
91 nvs_handle, id_to_key(id).c_str(), data.data(), data.size()));
92 if(err != ESP_OK)
93 return -1;
94
95 ESP_ERROR_CHECK_WITHOUT_ABORT(
96 err = nvs_set_u32(nvs_handle, last_id_name, id));
97 if(err != ESP_OK && err != ESP_ERR_NVS_NOT_FOUND)
98 return -1;
99
100 ESP_ERROR_CHECK_WITHOUT_ABORT(err = nvs_commit(nvs_handle));
101 if(err != ESP_OK)
102 return -1;
103
104 return 0;
105 }
106
108 [[nodiscard]] int read(uint32_t id, std::span<uint8_t> &buffer) override
109 {
110 esp_err_t err;
111 std::size_t required_size = 0;
112
113 // query the required size
114 ESP_ERROR_CHECK_WITHOUT_ABORT(
115 err = nvs_get_blob(
116 nvs_handle, id_to_key(id).c_str(), nullptr, &required_size));
117 if(err != ESP_OK)
118 return -1;
119
120 if(required_size > buffer.size())
121 {
122 ESP_LOGE(TAG,
123 "Not enough space in the output buffer. Need %zu, got %zu",
124 required_size,
125 buffer.size());
126 return -1;
127 }
128
129 ESP_ERROR_CHECK_WITHOUT_ABORT(err = nvs_get_blob(nvs_handle,
130 id_to_key(id).c_str(),
131 buffer.data(),
132 &required_size));
133 if(err != ESP_OK)
134 return -1;
135
136 buffer = buffer.subspan(0, required_size);
137 return 0;
138 };
139

142 const std::span<uint8_t> &data) override

144 esp_err_t err;
145
146 ESP_ERROR_CHECK_WITHOUT_ABORT(
147 err = nvs_set_blob(

150 return -1;
151

155

160 [[nodiscard]] int destroy(uint32_t id) override
161 {
162 esp_err_t err;
163
164 ESP_ERROR_CHECK_WITHOUT_ABORT(
165 err = nvs_erase_key(nvs_handle, id_to_key(id).c_str()));
166 if(err != ESP_OK)
167 return -1;
168
169 ESP_ERROR_CHECK_WITHOUT_ABORT(err = nvs_commit(nvs_handle));
170 if(err != ESP_OK)
171 return -1;
172
173 return 0;
174 };

176 void for_each(etl::delegate<void(uint32_t, const std::span<uint8_t> &)>
177 callback) override
178 {
179 for(const auto &entry : *this)
180 {
181 std::array<uint8_t, READ_BUFFER_SIZE> memory;
182 auto buffer = std::span<uint8_t>(memory);
183
184 auto id = key_to_id(entry.key);
185 auto rc = read(id, buffer);
186 if(rc != 0)
187 {
188 ESP_LOGE(TAG, "Failed to read preset at key %s", entry.key);
189 continue;
190 }
191 callback(id, buffer);
192 }
193 }

195private:
196 struct Iterator
197 {
198 using difference_type = int;
199 using value_type = nvs_entry_info_t;
200
201 // An end iterator
202 Iterator()
203 : part_name(nullptr),
204 namespace_name(nullptr),
205 type(NVS_TYPE_BLOB)
206 {
207 }
208
209 Iterator(const char *part_name,
210 const char *namespace_name,
211 nvs_type_t type)
212 : part_name(part_name),
213 namespace_name(namespace_name),
214 type(type)
215 {
216 esp_err_t rc =
217 nvs_entry_find(part_name, namespace_name, type, &self);
218 if(rc == ESP_ERR_NVS_NOT_FOUND)
219 {
220 assert(self == nullptr);
221 }
222 else if(rc != ESP_OK)
223 {
224 assert(self == nullptr);
225 ESP_LOGE(TAG,
226 "nvs_entry_find failed: %d %s",
227 rc,
228 esp_err_to_name(rc));
229 }
230 }
231 ~Iterator() { nvs_release_iterator(self); }
232
233 Iterator(const Iterator &other)
234 : part_name(other.part_name),
235 namespace_name(other.namespace_name),
236 type(other.type)
237 {
238 int rc = nvs_entry_find(part_name, namespace_name, type, &self);
239
240 if(rc == ESP_ERR_NVS_NOT_FOUND)
241 {
242 assert(self == nullptr);
243 return;
244 }
245 else if(rc != ESP_OK)
246 {
247 assert(self == nullptr);
248 ESP_LOGE(TAG,
249 "nvs_entry_find failed: %d %s",
250 rc,
251 esp_err_to_name(rc));
252 return;
253 }
254
255 for(i = 0; i < other.i; i++)
256 {
257 ESP_ERROR_CHECK(nvs_entry_next(&self));
258 }
259 }
260
261 Iterator &operator=(const Iterator &other)
262 {
263 if(this == &other)
264 return *this;
265
266 i = 0;
267 part_name = other.part_name;
268 namespace_name = other.namespace_name;
269 type = other.type;
270 nvs_release_iterator(self);
271
272 ESP_ERROR_CHECK(
273 nvs_entry_find(part_name, namespace_name, type, &self));
274 for(i = 0; i < other.i; i++)
275 {
276 ESP_ERROR_CHECK(nvs_entry_next(&self));
277 }
278
279 return *this;
280 }
281
282 nvs_entry_info_t operator*() const
283 {
284 nvs_entry_info_t info;
285 ESP_ERROR_CHECK(nvs_entry_info(self, &info));
286 return info;
287 };

289 Iterator &operator++()
290 {
291 next();
292 return *this;
293 }

295 Iterator operator++(int)
296 {
297 Iterator tmp = *this;
298 next();
299 return tmp;
300 }

302 friend bool operator==(const Iterator &left, const Iterator &right)
303 {
304 // all end iterators are equal
305 if(left.self == nullptr && right.self == nullptr)
306 return true;
307
308 // end iterator is not equal to any valid one
309 if(left.self == nullptr || right.self == nullptr)
310 return false;
311
312 // pointers to the same iterator are equal
313 if(left.self == right.self)
314 return true;
315
316 // if partitions are different, then the iterators can't be equal
317 auto left_partition = std::basic_string_view(left.part_name);
318 auto right_partition = std::basic_string_view(right.part_name);
319 if(left_partition != right_partition)
320 return false;
321
322 // if namespaces are different, then the iterators can't be equal
323 auto left_namespace =
324 std::basic_string_view((*left).namespace_name);
325 auto right_namespace =
326 std::basic_string_view((*right).namespace_name);
327 if(left_namespace != right_namespace)
328 return false;
329
330 auto left_key = std::basic_string_view((*left).key);
331 auto right_key = std::basic_string_view((*right).key);
332 if(left_key != right_key)
333 return false;
334
335 // type doesn't matter, these are pointing to the same thing
336 return true;
337 }
338
339 private:
340 void next()
341 {
342 i++;
343 int rc = nvs_entry_next(&self);
344 if(rc == ESP_ERR_NVS_NOT_FOUND)
345 {
346 assert(self == nullptr);
347 return;
348 }
349 else if(rc != ESP_OK)
350 {
351 ESP_LOGE(TAG,
352 "nvs_entry_next failed: %d %s",
353 rc,
354 esp_err_to_name(rc));
355 }
356 }
357
358 const char *part_name;
359 const char *namespace_name;
360 nvs_type_t type;
361 int i = 0;
362 nvs_iterator_t self = nullptr;
363 };
364
365 Iterator begin() { return {part_name, namespace_name, NVS_TYPE_BLOB}; }
366
367 Iterator end() { return Iterator(); }
368
369 const char *const part_name;
370 const char *const namespace_name;
371 nvs_handle_t nvs_handle = 0;

373 static constexpr char last_id_name[] = "last_id";
374 static constexpr char TAG[] = "esp_crud";

376 static_assert(std::input_iterator<EspCrud::Iterator>);
377};
