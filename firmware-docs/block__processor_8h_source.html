27template <size_t N, typename ChildProcessor>
28class BlockProcessor : public juce::dsp::ProcessorBase
29{
30public:
31 explicit BlockProcessor(ChildProcessor a_child) : child{std::move(a_child)}
32 {
33 }
34
35 void prepare(const juce::dsp::ProcessSpec &spec) override
36 {
37 reset();
38
39 auto childSpec = spec;
40 childSpec.maximumBlockSize = N;
41 child.prepare(childSpec);
42 };
43
44 void reset() override
45 {
46 buffers.clear();
47 child.reset();
48 };
49
50 void
51 process(const juce::dsp::ProcessContextReplacing<float> &context) override
52 {
53 auto io_block = context.getOutputBlock();
54 auto samples_left = io_block.getNumSamples();
55 size_t output_done_count = 0;
56
57 while(samples_left > 0)
58 {
59 auto space_left_in_write_buffer = N - write_size;
60 auto samples_processed_this_iteration =
61 juce::jmin(space_left_in_write_buffer, samples_left);
62 samples_left -= samples_processed_this_iteration;
63
64 // We must never overflow the buffer
65 jassert(write_size + samples_processed_this_iteration <= N);
66
67 // Add input samples to the internal buffer
68 auto write_buffer = buffers.getWritePointer(!read_buffer_channel);
69 std::copy(io_block.getChannelPointer(0) + output_done_count,
70 io_block.getChannelPointer(0) + output_done_count +
71 samples_processed_this_iteration,
72 &write_buffer[write_size]);
73 write_size += samples_processed_this_iteration;
74
75 // Take outputs samples from the internal buffer
76 auto read_buffer = buffers.getReadPointer(read_buffer_channel);
77 std::copy(
78 &read_buffer[read_size],
79 &read_buffer[read_size + samples_processed_this_iteration],
80 &io_block.getChannelPointer(0)[output_done_count]);
81 output_done_count += samples_processed_this_iteration;
82 read_size += samples_processed_this_iteration;
83
84 jassert(read_size == write_size);
85
86 // After pushing, the buffer might now be full. If it is, then we
87 // let the block processor process it.
88 if(write_size == N)
89 {
90 auto completeBlock =
91 block.getSingleChannelBlock(!read_buffer_channel);
92 child.process(completeBlock);
93 read_size = 0;
94 write_size = 0;
95 read_buffer_channel = !read_buffer_channel;
96 }
97 }
98 };
99
100private:
101 ChildProcessor child;
102 juce::AudioBuffer<float> buffers{2, N};
103 juce::dsp::AudioBlock<float> block{buffers};
104 size_t read_buffer_channel = 0;
105 size_t write_size = 0;
106 size_t read_size = 0;
107};
