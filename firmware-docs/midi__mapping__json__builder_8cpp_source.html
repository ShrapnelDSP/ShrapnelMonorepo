108template<>
109rapidjson::Value to_json(rapidjson::Document &document, const MappingApiMessage &object)
110{
111 rapidjson::Value json;
112
113 auto visitor = [&](const auto &message) {
114 using T = std::decay_t<decltype(message)>;
115
116 if constexpr (std::is_same_v<T, CreateResponse> ||
117 std::is_same_v<T, GetResponse>) {
118 json = to_json(document, message);
119 } else {
120 ESP_LOGE(TAG, "No handler registered for message");
121 }
122
123 json.AddMember("messageType", rapidjson::StringRef(get_message_type<T>()), document.GetAllocator());
124 };
125
126 std::visit(visitor, object);
127
128 return json;
129}
