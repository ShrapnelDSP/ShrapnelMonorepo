33template<>
34rapidjson::Value to_json(rapidjson::Document &document, const Message &object)
35{
36 rapidjson::Value json;
37 json.SetObject();
38
39 json.AddMember("channel", object.channel, document.GetAllocator());
40
41 std::visit(
42 [&](auto &parameters)
43 {
44 using T = std::decay_t<decltype(parameters)>;
45
46 if constexpr(std::is_same_v<T, Message::NoteOn>)
47 {
48 json.AddMember("runtimeType",
49 rapidjson::StringRef("noteOn"),
50 document.GetAllocator());
51 json.AddMember(
52 "note", parameters.note, document.GetAllocator());
53 json.AddMember(
54 "velocity", parameters.velocity, document.GetAllocator());
55 }
56 else if constexpr(std::is_same_v<T, Message::NoteOff>)
57 {
58 json.AddMember("runtimeType",
59 rapidjson::StringRef("noteOff"),
60 document.GetAllocator());
61 json.AddMember(
62 "note", parameters.note, document.GetAllocator());
63 json.AddMember(
64 "velocity", parameters.velocity, document.GetAllocator());
65 }
66 else if constexpr(std::is_same_v<T, Message::ControlChange>)
67 {
68 json.AddMember("runtimeType",
69 rapidjson::StringRef("controlChange"),
70 document.GetAllocator());
71 json.AddMember(
72 "control", parameters.control, document.GetAllocator());
73 json.AddMember(
74 "value", parameters.value, document.GetAllocator());
75 }
76 else if constexpr(std::is_same_v<T, Message::ProgramChange>)
77 {
78 json.AddMember("runtimeType",
79 rapidjson::StringRef("programChange"),
80 document.GetAllocator());
81 json.AddMember(
82 "number", parameters.number, document.GetAllocator());
83 }
84 },
85 object.parameters);
86
87 return json;
88}

90template<>
91rapidjson::Value to_json(rapidjson::Document &document, const Mapping &object)
92{
93 rapidjson::Value json;
94 json.SetObject();
95
96 rapidjson::Value midi_channel;
97 midi_channel = object.midi_channel;
98 json.AddMember("midi_channel", midi_channel, document.GetAllocator());
99
100 rapidjson::Value cc_number;
101 cc_number = object.cc_number;
102 json.AddMember("cc_number", cc_number, document.GetAllocator());
103
104 rapidjson::Value mode;
105 mode.SetString(object.mode == Mapping::Mode::TOGGLE ? "toggle"
106 : "parameter",
107 document.GetAllocator());
108 json.AddMember("mode", mode, document.GetAllocator());
109
110 rapidjson::Value parameter_id;
111 parameter_id.SetString(object.parameter_name.data(), object.parameter_name.size(), document.GetAllocator());
112 json.AddMember("parameter_id", parameter_id, document.GetAllocator());
113
114 return json;
115}

179template<>
180rapidjson::Value to_json(rapidjson::Document &document, const MappingApiMessage &object)
181{
182 rapidjson::Value json;
183
184 auto visitor = [&](const auto &message) {
185 using T = std::decay_t<decltype(message)>;
186
187 if constexpr (std::is_same_v<T, CreateResponse> ||
188 std::is_same_v<T, GetResponse> ||
189 std::is_same_v<T, MessageReceived>) {
190 json = to_json(document, message);
191 } else {
192 ESP_LOGE(TAG, "No handler registered for message");
193 }
194
195 json.AddMember("messageType", rapidjson::StringRef(get_message_type<T>()), document.GetAllocator());
196 };
197
198 std::visit(visitor, object);
199
200 return json;
201}

203template<>
204rapidjson::Value to_json(rapidjson::Document &document, const etl::imap<Mapping::id_t, Mapping> &object)
205{
206 rapidjson::Value json;
207 json.SetObject();
208
209 for(const auto &entry : object)
210 {
211 rapidjson::Value mapping = to_json(document, entry.second);
212 auto id = to_json(document, entry.first);
213 json.AddMember(id, mapping, document.GetAllocator());
214 }
215
216 return json;
217}
