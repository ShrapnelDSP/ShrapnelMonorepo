142template <std::size_t MAX_PARAMETERS>
143class ParameterObserver final : public parameters::ParameterObserver
144{
145public:
146 explicit ParameterObserver(
147 std::shared_ptr<persistence::Storage> a_persistence)
148 : is_save_throttled{true},
149 persistence{a_persistence},
150 timer{"param save throttle",
151 pdMS_TO_TICKS(10'000),
152 false,
153 etl::delegate<void(void)>::create<
154 ParameterObserver<MAX_PARAMETERS>,
155 &ParameterObserver<MAX_PARAMETERS>::timer_callback>(*this)}
156 {
157 }
158
159 void
160 notification(std::pair<const parameters::id_t &, float> parameter) override
161 {
162 auto &[id, value] = parameter;
163 ESP_LOGI(
164 TAG, "notified about parameter change %s %f", id.data(), value);
165 if(!updated_parameters.available())
166 {
167 ESP_LOGE(TAG, "no space available");
168 return;
169 }
170
171 updated_parameters[id] = value;
172
173 if(!timer.is_active())
174 {
175 if(pdPASS != timer.start(pdMS_TO_TICKS(5)))
176 {
177 ESP_LOGE(TAG, "Failed to start parameter observer timer");
178 }
179 }
180 }
181
182 void loop()
183 {
184 if(!is_save_throttled.test_and_set())
185 {
186 persist_parameters();
187 ESP_LOGI(TAG, "Parameters saved to NVS");
188 }
189 }
190
191private:
192 void timer_callback()
193 {
194 is_save_throttled.clear();
195 is_save_throttled.notify_all();
196 }
197
198 void persist_parameters()
199 {
200 for(const auto &param : updated_parameters)
201 {
202 rapidjson::Document document;
203
204 auto parameter_data = midi::to_json(document, param.second);
205 document.Swap(parameter_data);
206
207 rapidjson::StringBuffer buffer;
208 rapidjson::Writer writer{buffer};
209 document.Accept(writer);
210
211 persistence->save(param.first.data(), buffer.GetString());
212 }
213
214 updated_parameters.clear();
215 }
216
217 std::atomic_flag is_save_throttled;
218 std::shared_ptr<persistence::Storage> persistence;
219 os::Timer timer;
220 etl::map<parameters::id_t, float, MAX_PARAMETERS> updated_parameters;
221};

382 void loop()
383 {
384 {
385 auto byte = midi_uart->get_byte(0);
386 while(byte.has_value())
387 {
388 ESP_LOGI(TAG, "midi got byte 0x%02x", *byte);
389 midi_decoder->decode(*byte);
390 byte = midi_uart->get_byte(0);
391 }
392 }
393
394 if(AppMessage message; in_queue.receive(&message, 0))
395 {
396 auto fd = message.second;
397
398 std::visit([this, fd](const auto &m) { handle_message(m, fd); },
399 message.first);
400 }
401
402 if(!clipping_throttle_timer.is_active())
403 {
404 if(!events::input_clipped.test_and_set())
405 {
406 ESP_LOGI(TAG, "input was clipped");
407 send_message({events::InputClipped{}, std::nullopt});
408 BaseType_t rc =
409 clipping_throttle_timer.start(pdMS_TO_TICKS(10));
410 if(rc != pdPASS)
411 {
412 ESP_LOGE(TAG, "Failed to start clipping throttle timer");
413 }
414 }
415
416 if(!events::output_clipped.test_and_set())
417 {
418 ESP_LOGI(TAG, "output was clipped");
419 send_message({events::OutputClipped{}, std::nullopt});
420 BaseType_t rc =
421 clipping_throttle_timer.start(pdMS_TO_TICKS(10));
422 if(rc != pdPASS)
423 {
424 ESP_LOGE(TAG, "Failed to start clipping throttle timer");
425 }
426 }
427 }
428
429 parameter_observer.loop();
430
431 if(!is_midi_notify_waiting.test_and_set() &&
432 last_midi_message.has_value() &&
433 last_notified_midi_message != *last_midi_message)
434 {
435 last_notified_midi_message = *last_midi_message;
436
437 send_message(
438 {midi::MessageReceived{*last_midi_message}, std::nullopt});
439 }
440 }

442private:
443 void handle_message(const auto &app_message, std::optional<int> fd)
444 {
445 using AppMessageT = std::decay_t<decltype(app_message)>;
446
447 if constexpr(std::is_same_v<AppMessageT, parameters::ApiMessage>)
448 {
449 if(!fd.has_value())
450 {
451 ESP_LOGE(TAG, "Must always have fd");
452 }
453
454 cmd_handling->dispatch(app_message, *fd);
455 }
456 else if constexpr(std::is_same_v<AppMessageT, midi::MappingApiMessage>)
457 {
458 std::scoped_lock lock{midi_mutex};
459
460 auto response = std::visit(
461 [&](const auto &midi_mapping_message)
462 -> std::optional<midi::MappingApiMessage>
463 {
464 using MidiMappingMessageT =
465 std::decay_t<decltype(midi_mapping_message)>;
466
467 if constexpr(std::is_same_v<MidiMappingMessageT,
468 midi::GetRequest>)
469 {
470 auto mappings = midi_mapping_manager->get();
471 return midi::GetResponse{mappings};
472 }
473 else if constexpr(std::is_same_v<MidiMappingMessageT,
474 midi::CreateRequest>)
475 {
476 auto rc = midi_mapping_manager->create(
477 midi_mapping_message.mapping);
478 if(rc == 0)
479 {
480 return midi::CreateResponse{
481 midi_mapping_message.mapping};
482 }
483 }
484 else if constexpr(std::is_same_v<MidiMappingMessageT,
485 midi::Update>)
486 {
487 // return code ignored, as there is no way
488 // to indicate the error to the frontend
489 (void)midi_mapping_manager->update(
490 midi_mapping_message.mapping);
491 }
492 else if constexpr(std::is_same_v<MidiMappingMessageT,
493 midi::Remove>)
494 {
495 midi_mapping_manager->remove(midi_mapping_message.id);
496 }
497 else
498 {
499 ESP_LOGE(TAG, "Unhandled message type");
500 }
501
502 return std::nullopt;
503 },
504 app_message);
505
506 if(response.has_value())
507 {
508 send_message({*response, std::nullopt});
509 }
510 }
511 else if constexpr(std::is_same_v<AppMessageT,
512 presets::PresetsApiMessage>)
513 {
514 auto response = std::visit(
515 [&](const auto &presets_message)
516 -> std::optional<presets::PresetsApiMessage>
517 {
518 using PresetsMessageT =
519 std::decay_t<decltype(presets_message)>;
520
521 if constexpr(std::is_same_v<PresetsMessageT,
522 presets::Initialise>)
523 {
524 presets_manager->for_each(
525 [this](presets::id_t id,
526 const presets::PresetData &preset)
527 {
528 send_message({presets::Notify{
529 .id = id,
530 .preset = preset,
531 },
532 std::nullopt});
533 });
534 }
535 else if constexpr(std::is_same_v<PresetsMessageT,
536 presets::Create>)
537 {
538 presets::id_t id;
539 int rc =
540 presets_manager->create(presets_message.preset, id);
541 if(rc != 0)
542 {
543 ESP_LOGE(TAG, "Failed to create preset");
544 return std::nullopt;
545 }
546 return presets::Notify{
547 .id{id},
548 .preset{presets_message.preset},
549 };
550 }
551 else if constexpr(std::is_same_v<PresetsMessageT,
552 presets::Update>)
553 {
554 int rc = presets_manager->update(
555 presets_message.id, presets_message.preset);
556 if(rc != 0)
557 {
558 ESP_LOGE(TAG, "Failed to update preset");
559 return std::nullopt;
560 }
561
562 return presets::Notify{
563 .id{presets_message.id},
564 .preset{presets_message.preset},
565 };
566 }
567 else if constexpr(std::is_same_v<PresetsMessageT,
568 presets::Delete>)
569 {
570 int rc = presets_manager->remove(presets_message.id);
571 if(rc != 0)
572 {
573 ESP_LOGE(TAG, "Failed to remove preset");
574 return std::nullopt;
575 }
576 }
577 else
578 {
579 ESP_LOGE(TAG, "Unhandled message type");
580 }
581
582 return std::nullopt;
583 },
584 app_message);
585
586 if(response.has_value())
587 {
588 send_message({*response, std::nullopt});
589 }
590 }
591 else if constexpr(std::is_same_v<
592 AppMessageT,
593 selected_preset::SelectedPresetApiMessage>)
594 {
595 auto response = std::visit(
596 [this](const auto &message)
597 -> std::optional<selected_preset::SelectedPresetApiMessage>
598 { return handle_selected_preset_message(message); },
599 app_message);
600
601 if(response.has_value())
602 {
603 send_message({*response, std::nullopt});
604 }
605 }
606 }
