62etl::string_stream &operator<<(etl::string_stream &out,
63 const PresetsApiMessage &self)
64{
65 auto print = [&](const auto &message)
66 {
67 using T = std::decay_t<decltype(message)>;
68
69 if constexpr(std::is_same_v<T, Initialise>)
70 {
71 out << "<Initialise>" << message;
72 }
73 else if constexpr(std::is_same_v<T, Notify>)
74 {
75 out << "<Notify>" << message;
76 }
77 else if constexpr(std::is_same_v<T, Create>)
78 {
79 out << "<Create>" << message;
80 }
81 else if constexpr(std::is_same_v<T, Update>)
82 {
83 out << "<Update>" << message;
84 }
85 else if constexpr(std::is_same_v<T, Delete>)
86 {
87 out << "<Delete>" << message;
88 }
89 else
90 {
91 out << "Unknown";
92 }
93 };
94
95 std::visit(print, self);
96 return out;
97}
