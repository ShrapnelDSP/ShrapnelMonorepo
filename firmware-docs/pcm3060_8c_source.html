37esp_err_t _i2c_write(uint8_t reg_adr, uint8_t data)
38{
39 esp_err_t err = i2c_write_byte(_port, _address, reg_adr, data);
40
41 if(err != ESP_OK)
42 {
43 ESP_LOGE(TAG, "Write to reg 0x%02x failed", reg_adr);

49void pcm3060_init(i2c_port_t port, bool adr)

55esp_err_t pcm3060_power_up(void)
56{
57 esp_err_t err;
58
59 if(!_initialised())
60 {
61 return ESP_FAIL;
62 }
63
64 err = _i2c_write(0x43, 0x80);//use clock1 for DAC, slave mode, 24-bit I2S
65 if(err != ESP_OK)
66 {
67 return err;
68 }
69
70 err = _i2c_write(0x40, 0xC0);//differential output, power save release
71 return err;
72}

74esp_err_t pcm3060_power_down(void)
75{
76 if(!_initialised())
77 {
78 return ESP_FAIL;
79 }
80
81 return _i2c_write(0x40, 0xF1);//single ended output, power save enable
82}
