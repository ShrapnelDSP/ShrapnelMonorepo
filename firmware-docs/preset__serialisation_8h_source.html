86{

102 preset.noise_gate_bypass = parameters.noise_gate_bypass * 1000;
103 preset.chorus_rate = parameters.chorus_rate * 1000;
104 preset.chorus_depth = parameters.chorus_depth * 1000;
105 preset.chorus_mix = parameters.chorus_mix * 1000;
106 preset.chorus_bypass = parameters.chorus_bypass * 1000;
107 preset.wah_position = parameters.wah_position * 1000;
108 preset.wah_vocal = parameters.wah_vocal * 1000;
109 preset.wah_bypass = parameters.wah_bypass * 1000;
110
111 auto packed_size = preset_parameters__get_packed_size(&preset);
112 if(packed_size > buffer.size())
113 {
114 return -1;
115 }
116
117 preset_parameters__pack(&preset, buffer.data());
118 buffer = buffer.subspan(0, packed_size);
119 return 0;

146 parameters_proto.chorus_rate = preset.parameters.chorus_rate * 1000;
147 parameters_proto.chorus_depth = preset.parameters.chorus_depth * 1000;
148 parameters_proto.chorus_mix = preset.parameters.chorus_mix * 1000;
149 parameters_proto.chorus_bypass = preset.parameters.chorus_bypass * 1000;
150 parameters_proto.wah_position = preset.parameters.wah_position * 1000;
151 parameters_proto.wah_vocal = preset.parameters.wah_vocal * 1000;
152 parameters_proto.wah_bypass = preset.parameters.wah_bypass * 1000;
153
154 struct Preset preset_proto = PRESET__INIT;
155 preset__init(&preset_proto);
156
157 preset_proto.name = const_cast<char *>(preset.name.c_str());
158 preset_proto.parameters = &parameters_proto;
159
160 auto packed_size = preset__get_packed_size(&preset_proto);
161 if(packed_size > buffer.size())
162 {
163 return -1;
164 }
165
166 preset__pack(&preset_proto, buffer.data());
167 buffer = buffer.subspan(0, packed_size);
168 return 0;
169}

171inline int deserialise_preset(const std::span<uint8_t> &buffer,
172 PresetData &preset)
173{
