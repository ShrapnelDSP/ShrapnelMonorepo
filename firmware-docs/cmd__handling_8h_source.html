57 void dispatch(const ApiMessage &a_message, int fd)
58 {
59#if !defined(TESTING)
60 ESP_LOGI(TAG,
61 "%s stack %d",
62 __FUNCTION__,
63 uxTaskGetStackHighWaterMark(NULL));
64#endif
65
66 std::visit(
67 [&](const auto &message) -> void
68 {
69 using T = std::decay_t<decltype(message)>;
70
71 if constexpr(std::is_same_v<T, Update>)
72 {
73 parameter_update(message, fd);
74 }
75 else if constexpr(std::is_same_v<T, Initialise>)
76 {
77 initialise_parameters();
78 }
79 else
80 {
81 ESP_LOGE(TAG, "Unhandled message type");
82 }
83 },
84 a_message);
85 }
86
87private:
88 void parameter_update(const Update &message, int fd)
89 {
90 int rc = param->update(message.id, message.value);
91 if(rc != 0)
92 {
93 ESP_LOGE(TAG,
94 "Failed to update parameter (%s) with value %f",
95 message.id.data(),
96 message.value);
97 }
98
99 send_message(message, fd);
100 }
101
102 void initialise_parameters()
103 {
104 for(const auto &[key, value] : *param)
105 {
106 Update message = {
107 .id{id_t{key}},
108 .value{value->get()},
109 };
110
111 send_message(message, std::nullopt);
112 }
113 }
114
115 std::shared_ptr<AudioParametersT> param;
116 SendMessageCallback send_message;
117
118 static inline const char *TAG = "cmd_handling";
119};
