106 void dispatch(const ApiMessage &a_message, int fd)
107 {
108#if !defined(TESTING)
109 ESP_LOGI(TAG, "%s stack %d", __FUNCTION__, uxTaskGetStackHighWaterMark(NULL));
110#endif
111
112 std::visit(
113 [&](const auto &message) -> void
114 {
115 using T = std::decay_t<decltype(message)>;
116
117 if constexpr(std::is_same_v<T, Update>)
118 {
119 parameter_update(message, fd);
120 }
121 else if constexpr(std::is_same_v<T, Initialise>)
122 {
123 initialise_parameters();
124 }
125 else
126 {
127 ESP_LOGE(TAG, "Unhandled message type");
128 }
129 },
130 a_message);
131 }
132
133 private:
134 void parameter_update(const Update &message, int fd)
135 {
136 int rc = param->update(message.id, message.value);
137 if(rc != 0)
138 {
139 ESP_LOGE(TAG, "Failed to update parameter (%s) with value %f", message.id.data(), message.value);
140 }
141
142 send_message(message, fd);
143 }
144
145 void initialise_parameters()
146 {
147 for(const auto& [key, value] : *param)
148 {
149 Update message = {
150 .id{id_t{key}},
151 .value{value->get()},
152 };
153
154 send_message(message, std::nullopt);
155 }
156 }
157
158 std::shared_ptr<AudioParametersT> param;
159 SendMessageCallback send_message;
160
161 static inline const char *TAG = "cmd_handling";
162};
