102 void work(void)
103 {
104 /* TODO should not leave these uninitialised */
105 char *message_type;
106 cJSON *type;
107
108 int ret = queue->receive(&message, portMAX_DELAY);
109 if(ret != pdTRUE)
110 {
111 ESP_LOGE(TAG, "Queue failed to receive");
112 return;
113 }
114
115#if !defined(TESTING)
116 ESP_LOGI(TAG, "%s stack %d", __FUNCTION__, uxTaskGetStackHighWaterMark(NULL));
117#endif
118 size_t message_size = sizeof(message.json);
119 assert(message_size <= INT_MAX);
120 (void)message_size;
121
122 ESP_LOGI(TAG, "received websocket message: %.*s",
123 static_cast<int>(sizeof(message.json)),
124 message.json);
125
126 json = cJSON_ParseWithLength(message.json, sizeof(message.json));
127 if(json == NULL)
128 {
129 ESP_LOGE(TAG, "json parsing failed");
130 goto done;
131 }
132
133 type = cJSON_GetObjectItemCaseSensitive(json, "messageType");
134 if(cJSON_IsString(type) && (type->valuestring != NULL))
135 {
136 message_type = type->valuestring;
137 }
138 else
139 {
140 ESP_LOGE(TAG, "error parsing messageType from json");
141 goto done;
142 }
143
144 if(0 == strcmp(message_type, "initialiseParameters"))
145 {
146 initialise_parameters();
147 }
148 else if(0 == strcmp(message_type, "parameterUpdate"))
149 {
150 parameter_update();
151 }
152 else
153 {
154 ESP_LOGE(TAG, "unknown message type (%s)", message_type);
155 }
156
157done:
158 cJSON_Delete(json);
159 }
160
161 private:
162 void parameter_update(void)
163 {
164 assert(json);
165
166 char *parsed_id;
167 cJSON *id = cJSON_GetObjectItemCaseSensitive(json, "id");
168 if(cJSON_IsString(id) && (id->valuestring != NULL))
169 {
170 parsed_id = id->valuestring;
171 }
172 else
173 {
174 ESP_LOGE(TAG, "error parsing id from json");
175 return;
176 }
177
178 float parsed_value;
179 cJSON *value = cJSON_GetObjectItemCaseSensitive(json, "value");
180 if(cJSON_IsNumber(value))
181 {
182 parsed_value = value->valuedouble;
183 }
184 else
185 {
186 ESP_LOGE(TAG, "error parsing value from json");
187 return;
188 }
189
190 int rc = param->update(parsed_id, parsed_value);
191 if(rc != 0)
192 {
193 ESP_LOGE(TAG, "Failed to update parameter (%s) with value %f", parsed_id, parsed_value);
194 }
195
196 event.send(message.json, message.fd);
197 }
198
199 void initialise_parameters(void)
200 {
201 assert(json);
202
203 Message output;
204
205 for(const auto& [key, value] : *param)
206 {
207 float tmp_f = value->get();
208
209 snprintf(output.json, sizeof(output.json),
210 "{\"id\":\"%s\",\"value\":%g}",
211 key.c_str(),
212 tmp_f);
213
214 event.send(output.json, -1);
215 }
216 }
