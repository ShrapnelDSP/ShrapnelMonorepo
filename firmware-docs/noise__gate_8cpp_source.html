112void gate_set_sample_rate(float a_sample_rate)
113{
114 std::array<float, 6> coeffs = {0};
115 dspal_biquad_design_lowpass(coeffs.data(), 10.f/a_sample_rate, M_SQRT1_2);
116
117 memmove(&coeffs.data()[4], &coeffs.data()[3], 2*sizeof(*coeffs.data()));
118 coeffs[3] = 1.f;
119
120 envelope_detect_filter->set_coefficients(coeffs);
121
122 sample_rate = a_sample_rate;
123}

125static float db_to_ratio(float db)
126{
127 return powf(10.f, db/20.f);
128}

130void gate_set_threshold(float a_threshold, float a_hysteresis)
131{
132 assert(a_hysteresis >= 0.f);
133
134 threshold = db_to_ratio(a_threshold);
135 threshold_low = threshold * db_to_ratio(-1.f * a_hysteresis);
136
137 assert(threshold_low <= threshold);
138}

155void gate_analyse(const float *buf, size_t sample_count)
156{
157 assert(sample_count <= buffer_size);
158
159 static float gain = 0;
160 static state_t next_state = INIT;
161 static int hold_count = 0;
162
163 // rectify the signal
164 for(size_t i = 0; i < sample_count; i++)
165 {
166 filter_buffer[i] = fabsf(buf[i]);
167 }
168
169 //low pass filter
170 envelope_detect_filter->process(filter_buffer, filter_buffer, sample_count);
171
172 // work out the gain
173 for(size_t i = 0; i < sample_count; i++)
174 {
175 switch(state)
176 {
177 case INIT:
178 //don't need to init hold_count, it will be initialised when
179 //leaving the attack state
180 gain = 0;
181 next_state = CLOSED;
182 break;
183 case CLOSED:
184 if(filter_buffer[i] > threshold)
185 {
186 next_state = ATTACK;
187 }
188 else
189 {
190 next_state = CLOSED;
191 }
192 break;
193 case ATTACK:
194 if(filter_buffer[i] > threshold)
195 {
196 gain = gain + 1.f/attack_samples;
197 if(gain > 1)
198 {
199 gain = 1;
200 hold_count = 0;
201 next_state = OPEN;
202 }
203 }
204 else
205 {
206 next_state = RELEASE;
207 }
208 break;
209 case OPEN:
210 if(filter_buffer[i] > threshold_low)
211 {
212 hold_count = 0;
213 next_state = OPEN;
214 }
215 else
216 {
217 hold_count = hold_count + 1;
218 if(hold_count > hold_samples)
219 {
220 next_state = RELEASE;
221 }
222 else
223 {
224 next_state = OPEN;
225 }
226 }
227 break;
228 case RELEASE:
229 if(filter_buffer[i] > threshold_low)
230 {
231 next_state = ATTACK;
232 }
233 else
234 {
235 gain = gain - 1.f/release_samples;
236 if(gain < 0)
237 {
238 gain = 0;
239 next_state = CLOSED;
240 }
241 }
242 break;
243 default:
244 next_state = INIT;
245 break;
246 }
247
248 state = next_state;
249 gain_buffer[i] = gain;
250 }
251}
