53int handle_midi(int argc, char *argv[])
54{
55 ESP_LOGI(TAG, "midi argc %d", argc);
56
57 for(int i = 0; i < argc; i++)
58 {
59 ESP_LOGI(TAG, "midi argv %d: %s", i, argv[i]);
60 }
61
62 int nerrors = arg_parse(argc, argv, (void **)(&arg_midi));
63 if(nerrors != 0)
64 {
65 char error_message_memory[1024] = {};
66 auto file =
67 fmemopen(error_message_memory, sizeof error_message_memory, "w");
68 arg_print_errors(file, arg_midi.end, "midi");
69
70 ESP_LOGE(TAG, "%s", error_message_memory);
71 return -1;
72 }
73
74 ESP_LOGI(TAG, "data: %s", arg_midi.data->sval[0]);
75
76 std::array<uint8_t, 128> buffer{};
77 size_t decoded_size = 0;
78 auto error =
79 cppcodec::base64_rfc4648::decode(buffer.data(),
80 buffer.size(),
81 arg_midi.data->sval[0],
82 strlen(arg_midi.data->sval[0]),
83 decoded_size);
84
85 if(!holds_alternative<std::monostate>(error))
86 {
87 ESP_LOGE(TAG, "base64 decode error");
88 return -1;
89 }
90
91 ESP_LOG_BUFFER_HEX_LEVEL(TAG, buffer.data(), decoded_size, ESP_LOG_INFO);
92
93 auto message =
94 api::from_bytes<midi::Message>(std::span(buffer.data(), decoded_size));
95 if(!message.has_value())
96 {
97 ESP_LOGE(TAG, "failed to decode MIDI message");
98 return -1;
99 }
100
101 etl::string<64> string;
102 etl::string_stream stream(string);
103
104 stream << message.value();
105
106 ESP_LOGI(TAG, "decoded %s", string.c_str());
107
108 instance->midi_message_callback(message.value());
109
110 return 0;
111}

113int handle_wifi(int argc, char *argv[])
114{
115 ESP_LOGI(TAG, "wifi argc %d", argc);
116
117 for(int i = 0; i < argc; i++)
118 {
119 ESP_LOGI(TAG, "wifi argv %d: %s", i, argv[i]);
120 }
121
122 int nerrors = arg_parse(argc, argv, (void **)(&arg_wifi));
123 if(nerrors != 0)
124 {
125 char error_message_memory[1024] = {};
126 auto file =
127 fmemopen(error_message_memory, sizeof error_message_memory, "w");
128 arg_print_errors(file, arg_wifi.end, "wifi");
129
130 ESP_LOGE(TAG, "%s", error_message_memory);
131 return -1;
132 }
133
134 if(arg_wifi.ssid->count != 1 || arg_wifi.password->count != 1)
135 {
136 ESP_LOGE(TAG, "Missing SSID or password");
137 return -1;
138 }
139
140 instance->wifi_setup_callback(arg_wifi.ssid->sval[0],
141 arg_wifi.password->sval[0]);
142
143 return 0;
144}

194void Console::handle_character(char c)
195{
196 auto needs_processing = embedded_cli_insert_char(&cli, c);
197 if(needs_processing)
198 {
199 const char *line = embedded_cli_get_line(&cli);
200 assert(line != nullptr);
201
202 int rc = 0;
203 int err = esp_console_run(line, &rc);
204 if(err != ESP_OK)
205 {
206 ESP_LOGE(
207 TAG, "esp_console_run failed %d %s", err, esp_err_to_name(err));
208 embedded_cli_prompt(&cli);
209 return;
210 }
211
212 if(rc != 0)
213 {
214 ESP_LOGE(TAG, "command returned %d", rc);
215 }
216
217 embedded_cli_prompt(&cli);
218 }
219}
