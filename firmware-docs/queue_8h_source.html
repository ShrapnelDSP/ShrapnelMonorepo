48template <typename T, std::size_t MAX_SIZE>
49 requires(MAX_SIZE > 0) && (MAX_SIZE < PTRDIFF_MAX)
50class Queue final : public QueueBase<T>
51{
52 using ticks = std::chrono::duration<TickType_t>;
53
54public:
55 Queue()
56 : QueueBase<T>(MAX_SIZE),
57 used_semaphore{0},
58 free_semaphore{MAX_SIZE}
59 {
60 }
61
62 [[nodiscard]] BaseType_t receive(T *out, TickType_t time_to_wait) override
63 {
64 // block until an item is available
65 bool success = used_semaphore.try_acquire_for(ticks(time_to_wait));
66 if(!success)
67 return errQUEUE_EMPTY;
68
69 // Receive the item
70 {
71 std::scoped_lock lock{mutex};
72 *out = queue.front();
73 queue.pop();
74 }
75 free_semaphore.release();
76 return pdPASS;
77 }
78
79 [[nodiscard]] BaseType_t send(const T *in, TickType_t time_to_wait) override
80 {
81 // block until a space is available
82 bool success = free_semaphore.try_acquire_for(ticks(time_to_wait));
83 if(!success)
84 return errQUEUE_FULL;
85
86 // Add item to queue, and notify receivers
87 {
88 std::scoped_lock lock{mutex};
89 queue.push(*in);
90 }
91 used_semaphore.release();
92 return pdPASS;
93 }
94
95private:
96 std::counting_semaphore<static_cast<std::ptrdiff_t>(MAX_SIZE)>
97 used_semaphore;
98 std::counting_semaphore<static_cast<std::ptrdiff_t>(MAX_SIZE)>
99 free_semaphore;
100 etl::deque<T, MAX_SIZE> queue_storage;
101 std::queue<T, etl::deque<T, MAX_SIZE>> queue;
102 std::mutex mutex;
103};
