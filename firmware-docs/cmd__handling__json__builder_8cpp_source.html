28template<>
29rapidjson::Value to_json(rapidjson::Document &document, const Update &object)
30{
31 rapidjson::Value json;
32 json.SetObject();
33
34 rapidjson::Value id;
35 id.SetString(object.id.data(), object.id.size(), document.GetAllocator());
36 json.AddMember("id", id, document.GetAllocator());
37
38 rapidjson::Value value;
39 value = object.value;
40 json.AddMember("value", value, document.GetAllocator());
41
42 return json;
43}

45template<>
46rapidjson::Value to_json(rapidjson::Document &document, const ApiMessage &object)
47{
48 static constexpr char TAG[] = "parameters::ApiMessage to json";
49 rapidjson::Value json;
50
51 auto visitor = [&](const auto &message) {
52 using T = std::decay_t<decltype(message)>;
53
54 if constexpr (std::is_same_v<T, Update>) {
55 json = to_json(document, message);
56 } else {
57 ESP_LOGE(TAG, "No handler registered for message");
58 }
59
60 json.AddMember("messageType", rapidjson::StringRef(get_message_type<T>()), document.GetAllocator());
61 };
62
63 std::visit(visitor, object);
64
65 return json;
66}
