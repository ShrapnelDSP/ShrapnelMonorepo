46std::optional<PresetData> PresetsManager::read(id_t id)
47{

56 PresetData preset;

97void PresetsManager::for_each(
98 etl::delegate<void(id_t, const PresetData &)> callback)
99{
100 for(const auto &entry : *storage)
101 {
102 auto id = key_to_id(entry.key);
103 auto preset = read(id);
104 if(!preset.has_value())
105 {
106 ESP_LOGE(TAG, "Failed to read preset at key %s", entry.key);
107 continue;
108 }
109 callback(id, *preset);
110 }
111}
