32int PresetsManager::create(const PresetData &preset, id_t &id_out)
33{
34 std::array<uint8_t, 100> buffer;
35 auto data = std::span<uint8_t, std::dynamic_extent>(buffer);
36
37 auto encoded = api::to_bytes(preset, data);
38 if(!encoded.has_value())
39 {
40 return -1;
41 }
42
43 int rc = storage->create(*encoded, id_out);
44 if(rc != 0)
45 {
46 return -1;
47 }
48
49 return 0;
50}

52int PresetsManager::read(id_t id, PresetData &preset)
53{
54 std::array<uint8_t, 100> buffer;

56 int rc = storage->read(id, data);

65 return -1;

104void PresetsManager::for_each(
105 etl::delegate<void(id_t, const PresetData &)> callback)
106{
107 storage->for_each(
108 [&callback](uint32_t id, std::span<uint8_t> data)
109 {
110 auto decoded = api::from_bytes<PresetData>(data);
111 if(!decoded.has_value())
112 {
113 return;
114 }
115
116 callback(id, *decoded);
117 });
