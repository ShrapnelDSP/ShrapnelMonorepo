177template <>
178int to_proto(const presets::PresetsApiMessage &message,
179 shrapnel_presets_Message &out)
180{
181 return std::visit(
182 [&out](const auto &message) -> int
183 {
184 using T = std::decay_t<decltype(message)>;
185 if constexpr(std::is_same_v<T, presets::Initialise>)
186 {
187 out.which_message = shrapnel_presets_Message_initialise_tag;
188 int rc = to_proto<shrapnel_presets_Initialise>(
189 message, out.message.initialise);
190 if(rc != 0)
191 {
192 return -1;
193 }
194 }
195 else if constexpr(std::is_same_v<T, presets::Notify>)
196 {
197 out.which_message = shrapnel_presets_Message_notify_tag;
198 int rc = to_proto<shrapnel_presets_Notify>(message,
199 out.message.notify);
200 if(rc != 0)
201 {
202 return -1;
203 }
204 }
205 else if constexpr(std::is_same_v<T, presets::Create>)
206 {
207 out.which_message = shrapnel_presets_Message_create_tag;
208 int rc = to_proto<shrapnel_presets_Create>(message,
209 out.message.create);
210 if(rc != 0)
211 {
212 return -1;
213 }
214 }
215 else if constexpr(std::is_same_v<T, presets::Update>)
216 {
217 out.which_message = shrapnel_presets_Message_update_tag;
218 int rc = to_proto<shrapnel_presets_Update>(message,
219 out.message.update);
220 if(rc != 0)
221 {
222 return -1;
223 }
224 }
225 else if constexpr(std::is_same_v<T, presets::Delete>)
226 {
227 out.which_message = shrapnel_presets_Message_remove_tag;
228 int rc = to_proto<shrapnel_presets_Remove>(message,
229 out.message.remove);
230 if(rc != 0)
231 {
232 return -1;
233 }
234 }
235 else
236 {
237 return -1;
238 }
239
240 return 0;
241 },
242 message);
243}

245template <>
246int from_proto(const shrapnel_presets_Message &message,
247 presets::PresetsApiMessage &out)
248{
249 switch(message.which_message)
250 {
251 case shrapnel_presets_Message_initialise_tag:
252 {
253 presets::Initialise tmp{};
254 from_proto<presets::Initialise>(message.message.initialise, tmp);
255 out = tmp;
256 return 0;
257 }
258 case shrapnel_presets_Message_notify_tag:
259 {
260 presets::Notify tmp{};
261 from_proto<presets::Notify>(message.message.notify, tmp);
262 out = tmp;
263 return 0;
264 }
265 case shrapnel_presets_Message_create_tag:
266 {
267 presets::Create tmp{};
268 from_proto<presets::Create>(message.message.create, tmp);
269 out = tmp;
270 return 0;
271 }
272 case shrapnel_presets_Message_update_tag:
273 {
274 presets::Update tmp{};
275 from_proto<presets::Update>(message.message.update, tmp);
276 out = tmp;
277 return 0;
278 }
279 case shrapnel_presets_Message_remove_tag:
280 {
281 presets::Delete tmp{};
282 from_proto<presets::Delete>(message.message.remove, tmp);
283 out = tmp;
284 return 0;
285 }
286 }
287
288 return -1;
289}

291template <>
292std::optional<std::span<uint8_t>> to_bytes(const shrapnel_presets_Preset &proto,
293 std::span<uint8_t> buffer)
294{
295 pb_ostream_t stream = pb_ostream_from_buffer(buffer.data(), buffer.size());
296
297 bool success = pb_encode(&stream, &shrapnel_presets_Preset_msg, &proto);
298 if(!success)
299 {
300 return std::nullopt;
301 }
302
303 return buffer.subspan(0, stream.bytes_written);
304}

308from_bytes(std::span<const uint8_t> buffer)
309{

312 bool success = pb_decode(&stream, &shrapnel_presets_Preset_msg, &unpacked);
313 if(!success)

316 }
317
318 return unpacked;
