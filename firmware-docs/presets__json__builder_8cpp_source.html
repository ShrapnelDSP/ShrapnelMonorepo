31template <>
32rapidjson::Value to_json(rapidjson::Document &document,
33 const ParametersData &object)
34{
35 uint8_t buffer[100] = {};
36 auto data = std::span<uint8_t, std::dynamic_extent>(buffer);
37
38 int rc = serialise_parameters(object, data);
39 if(rc != 0)
40 {
41 return {/* null */};
42 }
43
44 using base64_codec = cppcodec::base64_rfc4648;
45
46 std::array<char, 200> base64_buffer;
47 auto encoded_size = base64_codec::encoded_size(data.size());
48 if(encoded_size > base64_buffer.size())
49 {
50 return {/* null */};
51 }
52
53 auto error = base64_codec::encode(base64_buffer.data(),
54 base64_buffer.size(),
55 data.data(),
56 data.size(),
57 encoded_size);
58
59 if(!std::holds_alternative<std::monostate>(error))
60 {
61 return {/* null */};
62 }
63
64 auto json = rapidjson::Value{};
65 json.SetString(base64_buffer.data(), encoded_size, document.GetAllocator());
66 return json;
67}

98template <>
99rapidjson::Value to_json(rapidjson::Document &document,
100 const PresetsApiMessage &object)
101{
102 rapidjson::Value json;
103
104 auto visitor = [&](const auto &message)
105 {
106 using T = std::decay_t<decltype(message)>;
107
108 if constexpr(std::is_same_v<T, Notify>)
109 {
110 json = to_json(document, message);
111 }
112
113 json.AddMember("messageType",
114 rapidjson::StringRef(get_message_type<T>()),
115 document.GetAllocator());
116 };
117
118 std::visit(visitor, object);
119
120 return json;
121}
