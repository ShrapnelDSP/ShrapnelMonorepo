30template <>
31std::optional<ParametersData> from_json(const rapidjson::Value &json)
32{
33 using base64_codec = cppcodec::base64_rfc4648;
34 if(!json.IsString())
35 {
36 return std::nullopt;
37 }
38
39 std::array<uint8_t, 100> buffer{};
40 if(base64_codec::decoded_max_size(json.GetStringLength()) > buffer.size())
41 {
42 return std::nullopt;
43 }
44
45 size_t decoded_size = 0;
46 auto error = base64_codec::decode(buffer.data(),
47 buffer.size(),
48 json.GetString(),
49 json.GetStringLength(),
50 decoded_size);
51
52 if(!std::holds_alternative<std::monostate>(error))
53 {
54 return std::nullopt;
55 }
56
57 auto unpacked =
58 preset_parameters__unpack(nullptr, decoded_size, buffer.data());
59 if(unpacked == nullptr)
60 {
61 return std::nullopt;
62 }
63
64 auto out = ParametersData{
65 .amp_gain = unpacked->amp_gain / 1000.f,
66 .amp_channel = unpacked->amp_channel / 1000.f,
67 .bass = unpacked->bass / 1000.f,
68 .middle = unpacked->middle / 1000.f,
69 .treble = unpacked->treble / 1000.f,
70 .contour = unpacked->contour / 1000.f,
71 .volume = unpacked->volume / 1000.f,
72 .noise_gate_threshold = unpacked->noise_gate_threshold / 1000.f,
73 .noise_gate_hysteresis = unpacked->noise_gate_hysteresis / 1000.f,
74 .noise_gate_attack = unpacked->noise_gate_attack / 1000.f,
75 .noise_gate_hold = unpacked->noise_gate_hold / 1000.f,
76 .noise_gate_release = unpacked->noise_gate_release / 1000.f,
77 .noise_gate_bypass = unpacked->noise_gate_bypass / 1000.f,
78 .chorus_rate = unpacked->chorus_rate / 1000.f,
79 .chorus_depth = unpacked->chorus_depth / 1000.f,
80 .chorus_mix = unpacked->chorus_mix / 1000.f,
81 .chorus_bypass = unpacked->chorus_bypass / 1000.f,
82 .wah_position = unpacked->wah_position / 1000.f,
83 .wah_vocal = unpacked->wah_vocal / 1000.f,
84 .wah_bypass = unpacked->wah_bypass / 1000.f,
85 };
86
87 preset_parameters__free_unpacked(unpacked, nullptr);
88 return out;
89}

91template <>
92std::optional<PresetData> from_json(const rapidjson::Value &json)
93{
94 auto name = json.FindMember("name");
95 if(name == json.MemberEnd() || !name->value.IsString())
96 return std::nullopt;
97
98 auto parameters_json = json.FindMember("parameters");
99 if(parameters_json == json.MemberEnd())
100 return std::nullopt;
101 auto parameters = from_json<ParametersData>(parameters_json->value);
102 if(!parameters.has_value())
103 return std::nullopt;
104
105 return {{
106 .name{name->value.GetString()},
107 .parameters{*parameters},
108 }};
109}

163template <>
164std::optional<PresetsApiMessage> from_json(const rapidjson::Value &json)
165{
166 constexpr char TAG[] = "PresetsApiMessage from_json";
167
168 auto message_type_member = json.FindMember("messageType");
169 if(message_type_member == json.MemberEnd())
170 {
171 ESP_LOGE(TAG, "messageType is missing");
172 goto error;
173 }
174
175 if(!message_type_member->value.IsString())
176 {
177 ESP_LOGE(TAG, "messageType is not string");
178 goto error;
179 }
180
181 {
182 const etl::map<std::string,
183 std::function<std::optional<PresetsApiMessage>()>,
184 4>
185 lut{
186 {get_message_type<Initialise>(),
187 [&] { return from_json<Initialise>(json); }},
188 {get_message_type<Create>(),
189 [&] { return from_json<Create>(json); }},
190 {get_message_type<Update>(),
191 [&] { return from_json<Update>(json); }},
192 {get_message_type<Delete>(),
193 [&] { return from_json<Delete>(json); }},
194 };
195
196 const char *message_type = message_type_member->value.GetString();
197 if(auto f = lut.find(message_type); f != lut.end())
198 {
199 return f->second();
200 }
201 }
202
203error:
204 return std::nullopt;
205}
