67void log_i2s_event(const i2s_event_t &e)
68{
69 switch(e.type)
70 {
71 case I2S_EVENT_DMA_ERROR:
72 ESP_LOGE(TAG, "Something went wrong with DMA");
73 break;
74 case I2S_EVENT_TX_DONE:
75 case I2S_EVENT_RX_DONE:
76 //do nothing
77 break;
78 case I2S_EVENT_TX_Q_OVF:
79 ESP_LOGE(TAG, "TX queue overflowed");
80 break;
81 case I2S_EVENT_RX_Q_OVF:
82 ESP_LOGE(TAG, "RX queue overflowed");
83 break;
84 default:
85 ESP_LOGE(TAG, "Unhandled type: %d", e.type);
86 break;
87 }
88}

90static void i2s_processing_task(void *param)
91{
92 (void)param;
93 auto audio_param =
94 reinterpret_cast<shrapnel::parameters::AudioParametersBase *>(param);
95 size_t tx_rx_size;
96
97 i2s_config_t i2s_config = {
98 .mode = static_cast<i2s_mode_t>(I2S_MODE_MASTER | I2S_MODE_TX |
99 I2S_MODE_RX),
100 .sample_rate = SAMPLE_RATE,
101 .bits_per_sample = I2S_BITS_PER_SAMPLE_24BIT,
102 .channel_format = I2S_CHANNEL_FMT_RIGHT_LEFT,
103 .communication_format = I2S_COMM_FORMAT_STAND_I2S,
104 .intr_alloc_flags = ESP_INTR_FLAG_LEVEL1,
105 .dma_buf_count = DMA_BUF_COUNT,
106 .dma_buf_len = DMA_BUF_SIZE,
107 .use_apll = true,
108 .tx_desc_auto_clear =
109 true, //this will clear the tx buffer when we overrun
110 .fixed_mclk = 0, // Use mclk_multiple
111 .mclk_multiple = I2S_MCLK_MULTIPLE_384,
112 .bits_per_chan = I2S_BITS_PER_CHAN_DEFAULT,
113 };
114
115 i2s_pin_config_t pin_config = {.mck_io_num = I2S_MCK_IO,
116 .bck_io_num = I2S_BCK_IO,
117 .ws_io_num = I2S_WS_IO,
118 .data_out_num = I2S_DO_IO,
119 .data_in_num = I2S_DI_IO};
120
121 esp_err_t err = i2s_driver_install(static_cast<i2s_port_t>(I2S_NUM),
122 &i2s_config,
123 I2S_QUEUE_SIZE,
124 &i2s_queue);
125 if(err != ESP_OK)
126 {
127 ESP_LOGE(
128 TAG, "i2s_driver_install failed %d, %s", err, esp_err_to_name(err));
129 return;
130 }
131 ESP_LOGD(TAG, "queue address: %p", (void *)i2s_queue);
132
133 err = i2s_set_pin(static_cast<i2s_port_t>(I2S_NUM), &pin_config);
134 if(err != ESP_OK)
135 {
136 ESP_LOGE(TAG, "i2s_set_pin failed %d, %s", err, esp_err_to_name(err));
137 return;
138 }
139
140 auto processor_parameters = parameter_init(audio_param);
141 auto processor = shrapnel::AudioProcessor(processor_parameters);
142 processor.prepare(SAMPLE_RATE, DMA_BUF_SIZE);
143
144 /* Get the DMA buffers into their initial states
145 *
146 * The RX buffers should be all empty, and the TX buffers should all be
147 * full.
148 */
149 err = i2s_stop(static_cast<i2s_port_t>(I2S_NUM));
150 if(err != ESP_OK)
151 {
152 ESP_LOGE(TAG, "i2s_stop failed %d, %s", err, esp_err_to_name(err));
153 return;
154 }
155
156 size_t read_count;
157 do
158 {
159 err = i2s_read(static_cast<i2s_port_t>(I2S_NUM),
160 rx_buf,
161 DMA_BUF_SIZE,
162 &read_count,
163 100);
164 if(err != ESP_OK)
165 {
166 ESP_LOGE(TAG, "i2s_read failed %d, %s", err, esp_err_to_name(err));
167 return;
168 }
169 ESP_LOGI(TAG, "discarded %zu samples from RX buffer", read_count);
170 } while(read_count != 0);
171
172 err = i2s_zero_dma_buffer(static_cast<i2s_port_t>(I2S_NUM));
173 if(err != ESP_OK)
174 {
175 ESP_LOGE(TAG,
176 "i2s_zero_dma_buffer failed %d, %s",
177 err,
178 esp_err_to_name(err));
179 return;
180 }
181
182 err = i2s_start(static_cast<i2s_port_t>(I2S_NUM));
183 if(err != ESP_OK)
184 {
185 ESP_LOGE(TAG, "i2s_start failed %d, %s", err, esp_err_to_name(err));
186 return;
187 }
188
189 while(true)
190 {
191#if !defined(GENERATE_SINE_WAVE) && !defined(GENERATE_RAMP)
192 i2s_read(static_cast<i2s_port_t>(I2S_NUM),
193 rx_buf,
194 sizeof(rx_buf),
195 &tx_rx_size,
196 100 / portTICK_PERIOD_MS);
197 if(tx_rx_size != sizeof(rx_buf))
198 {
199 ESP_LOGE(TAG,
200 "Got the wrong number of bytes %zu/%zu",
201 tx_rx_size,
202 sizeof(rx_buf));
203 }
204#elif defined(GENERATE_SINE_WAVE)
205 int len = sizeof(rx_buf) / sizeof(rx_buf[0]);
206 for(int i = 0; i < len; i++)
207 {
208 rx_buf[i] =
209 float_to_int32(sinf(2 * ((float)M_PI) * i / (float)len));
210 }
211#elif defined(GENERATE_RAMP)
212 int len = sizeof(rx_buf) / sizeof(rx_buf[0]);
213 for(int i = 0; i < len; i++)
214 {
215 rx_buf[i] = (i - len / 2) * (INT32_MAX / (len / 2));
216 }
217#endif
218
219 process_samples(rx_buf, processor);
220
221 i2s_write(static_cast<i2s_port_t>(I2S_NUM),
222 rx_buf,
223 sizeof(rx_buf),
224 &tx_rx_size,
225 100 / portTICK_PERIOD_MS);
226 if(tx_rx_size != sizeof(rx_buf))
227 {
228 ESP_LOGE(TAG,
229 "Sent the wrong number of bytes %zu/%zu",
230 tx_rx_size,
231 sizeof(rx_buf));
232 }
233 }
234}

236esp_err_t i2s_setup(gpio_num_t profiling_gpio,
237 shrapnel::parameters::AudioParametersBase *audio_param)
238{
239 esp_err_t err;
240
241 g_profiling_gpio = profiling_gpio;
242 gpio_config_t io_conf = {
243 .pin_bit_mask = 1ULL << profiling_gpio,
244 .mode = GPIO_MODE_OUTPUT,
245 .pull_up_en = GPIO_PULLUP_DISABLE,
246 .pull_down_en = GPIO_PULLDOWN_DISABLE,
247 .intr_type = GPIO_INTR_DISABLE,
248 };
249
250 err = gpio_config(&io_conf);
251 if(err != ESP_OK)
252 {
253 ESP_LOGE(TAG, "gpio_config failed %d, %s", err, esp_err_to_name(err));
254 return err;
255 }
256 err = gpio_set_level(profiling_gpio, 0);
257 if(err != ESP_OK)
258 {
259 ESP_LOGE(
260 TAG, "gpio_set_level failed %d, %s", err, esp_err_to_name(err));
261 return err;
262 }
263
264 // This task must be pinned as CPU cycle count is used for profiling
265 int ret = xTaskCreatePinnedToCore(i2s_processing_task,
266 "i2s proc",
267 TASK_STACK,
268 audio_param,
269 TASK_PRIO,
270 NULL,
271 1);
272 if(ret != pdPASS)
273 {
274 ESP_LOGE(TAG, "Processing task create failed %d", ret);
275 }
276
277 return ESP_OK;
278}

281void process_samples(std::span<int32_t, 2 * DMA_BUF_SIZE> buf,
282 AudioProcessor &processor)
283{
284 assert(g_profiling_gpio != -1 && "I2S task has not been initialised");
285 static float fbuf[DMA_BUF_SIZE];
286
287 gpio_set_level(g_profiling_gpio, 1);
288 profiling_start();
289 /* Copy samples, skipping every second one. These correspond to the left
290 * channel of the ADC and DAC, which are not used here
291 * XXX not sure why the right channel is not at index 0 */
292 for(size_t i = 1; i < buf.size(); i += 2)
293 {
294 if(buf[i] == INT32_MIN || buf[i] == INT32_MAX)
295 {
296 shrapnel::events::input_clipped.clear();
297 shrapnel::events::input_clipped.notify_all();
298 }
299 fbuf[i / 2] = buf[i] / (float)INT32_MAX;
300 }
301
302 // call process on processor
303 processor.process(fbuf);
304
305 for(size_t i = 1; i < buf.size(); i += 2)
306 {
307 if(fbuf[i / 2] > 1.f || fbuf[i / 2] < -1.f)
308 {
309 shrapnel::events::output_clipped.clear();
310 shrapnel::events::output_clipped.notify_all();
311 }
312
313 if(fbuf[i / 2] != fbuf[i / 2])
314 {
315 ESP_LOGE(TAG, "Not a number at index %d", i);
316 assert(0);
317 }
318
319 buf[i] = float_to_int32(fbuf[i / 2]);
320 }
321 profiling_mark_stage(20);
322
323 profiling_stop();
324 gpio_set_level(g_profiling_gpio, 0);
325
326 ESP_LOGD(TAG, "stack: %d", uxTaskGetStackHighWaterMark(nullptr));
327}
