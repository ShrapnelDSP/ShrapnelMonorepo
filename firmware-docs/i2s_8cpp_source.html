53void log_i2s_event(const i2s_event_t &e)
54{
55 switch(e.type)
56 {
57 case I2S_EVENT_DMA_ERROR:
58 ESP_LOGE(TAG, "Something went wrong with DMA");
59 break;
60 case I2S_EVENT_TX_DONE:
61 case I2S_EVENT_RX_DONE:
62 //do nothing
63 break;
64 case I2S_EVENT_TX_Q_OVF:
65 ESP_LOGE(TAG, "TX queue overflowed");
66 break;
67 case I2S_EVENT_RX_Q_OVF:
68 ESP_LOGE(TAG, "RX queue overflowed");
69 break;
70 default:
71 ESP_LOGE(TAG, "Unhandled type: %d", e.type);
72 break;
73 }
74}

76static void i2s_processing_task(void *param)
77{
78 (void) param;
79 auto audio_param = reinterpret_cast<shrapnel::parameters::AudioParametersBase *>(param);
80
81 size_t tx_rx_size;
82
83 i2s_config_t i2s_config = {
84 .mode = static_cast<i2s_mode_t>(I2S_MODE_MASTER | I2S_MODE_TX | I2S_MODE_RX),
85 .sample_rate = SAMPLE_RATE,
86 .bits_per_sample = I2S_BITS_PER_SAMPLE_24BIT,
87 .channel_format = I2S_CHANNEL_FMT_RIGHT_LEFT,
88 .communication_format = I2S_COMM_FORMAT_STAND_I2S,
89 .intr_alloc_flags = ESP_INTR_FLAG_LEVEL1,
90 .dma_buf_count = DMA_BUF_COUNT,
91 .dma_buf_len = DMA_BUF_SIZE,
92 .use_apll = true,
93 .tx_desc_auto_clear = true, //this will clear the tx buffer when we overrun
94 .fixed_mclk = 0, // Use mclk_multiple
95 .mclk_multiple = I2S_MCLK_MULTIPLE_384,
96 .bits_per_chan = I2S_BITS_PER_CHAN_DEFAULT,
97 };
98
99 i2s_pin_config_t pin_config = {
100 .mck_io_num = I2S_MCK_IO,
101 .bck_io_num = I2S_BCK_IO,
102 .ws_io_num = I2S_WS_IO,
103 .data_out_num = I2S_DO_IO,
104 .data_in_num = I2S_DI_IO
105 };
106
107 esp_err_t err = i2s_driver_install(static_cast<i2s_port_t>(I2S_NUM), &i2s_config, I2S_QUEUE_SIZE, &i2s_queue);
108 if(err != ESP_OK)
109 {
110 ESP_LOGE(TAG, "i2s_driver_install failed %d, %s", err, esp_err_to_name(err));
111 return;
112 }
113 ESP_LOGD(TAG, "queue address: %p", (void*)i2s_queue);
114
115 err = i2s_set_pin(static_cast<i2s_port_t>(I2S_NUM), &pin_config);
116 if(err != ESP_OK)
117 {
118 ESP_LOGE(TAG, "i2s_set_pin failed %d, %s", err, esp_err_to_name(err));
119 return;
120 }
121
122
123 err = process_init(audio_param);
124 if(err != ESP_OK)
125 {
126 ESP_LOGE(TAG, "process_init failed %d, %s", err, esp_err_to_name(err));
127 return;
128 }
129
130 /* Get the DMA buffers into their initial states
131 *
132 * The RX buffers should be all empty, and the TX buffers should all be
133 * full.
134 */
135 err = i2s_stop(static_cast<i2s_port_t>(I2S_NUM));
136 if(err != ESP_OK)
137 {
138 ESP_LOGE(TAG, "i2s_stop failed %d, %s", err, esp_err_to_name(err));
139 return;
140 }
141
142 size_t read_count;
143 do {
144 err = i2s_read(static_cast<i2s_port_t>(I2S_NUM), rx_buf, DMA_BUF_SIZE, &read_count, 100);
145 if(err != ESP_OK)
146 {
147 ESP_LOGE(TAG, "i2s_read failed %d, %s", err, esp_err_to_name(err));
148 return;
149 }
150 ESP_LOGI(TAG, "discarded %zu samples from RX buffer", read_count);
151 }
152 while(read_count != 0);
153
154 err = i2s_zero_dma_buffer(static_cast<i2s_port_t>(I2S_NUM));
155 if(err != ESP_OK)
156 {
157 ESP_LOGE(TAG, "i2s_zero_dma_buffer failed %d, %s", err, esp_err_to_name(err));
158 return;
159 }
160
161 err = i2s_start(static_cast<i2s_port_t>(I2S_NUM));
162 if(err != ESP_OK)
163 {
164 ESP_LOGE(TAG, "i2s_start failed %d, %s", err, esp_err_to_name(err));
165 return;
166 }
167
168 while(1)
169 {
170#if !defined(GENERATE_SINE_WAVE) && !defined(GENERATE_RAMP)
171 i2s_read(static_cast<i2s_port_t>(I2S_NUM), rx_buf, sizeof(rx_buf), &tx_rx_size, 100/portTICK_PERIOD_MS);
172 if(tx_rx_size != sizeof(rx_buf))
173 {
174 ESP_LOGE(TAG, "Got the wrong number of bytes %zu/%zu", tx_rx_size, sizeof(rx_buf));
175 }
176#elif defined(GENERATE_SINE_WAVE)
177 int len = sizeof(rx_buf) / sizeof(rx_buf[0]);
178 for(int i = 0; i < len; i++)
179 {
180 rx_buf[i] = float_to_int32(sinf(2 * ((float)M_PI) * i/(float)len));
181 }
182#elif defined(GENERATE_RAMP)
183 int len = sizeof(rx_buf) / sizeof(rx_buf[0]);
184 for(int i = 0; i < len; i++)
185 {
186 rx_buf[i] = (i - len/2) * (INT32_MAX / (len/2));
187 }
188#endif
189
190 process_samples(rx_buf, sizeof(rx_buf) / sizeof(rx_buf[0]));
191
192 i2s_write(static_cast<i2s_port_t>(I2S_NUM), rx_buf, sizeof(rx_buf), &tx_rx_size, 100/portTICK_PERIOD_MS);
193 if(tx_rx_size != sizeof(rx_buf))
194 {
195 ESP_LOGE(TAG, "Sent the wrong number of bytes %zu/%zu", tx_rx_size, sizeof(rx_buf));
196 }
197 }
198}

200esp_err_t i2s_setup(gpio_num_t profiling_gpio, shrapnel::parameters::AudioParametersBase *audio_param)
201{
202 esp_err_t err;
203
204 g_profiling_gpio = profiling_gpio;
205 gpio_config_t io_conf = {
206 .pin_bit_mask = 1ULL << profiling_gpio,
207 .mode = GPIO_MODE_OUTPUT,
208 .pull_up_en = GPIO_PULLUP_DISABLE,
209 .pull_down_en = GPIO_PULLDOWN_DISABLE,
210 .intr_type = GPIO_INTR_DISABLE,
211 };
212
213 err = gpio_config(&io_conf);
214 if(err != ESP_OK)
215 {
216 ESP_LOGE(TAG, "gpio_config failed %d, %s", err, esp_err_to_name(err));
217 return err;
218 }
219 err = gpio_set_level(profiling_gpio, 0);
220 if(err != ESP_OK)
221 {
222 ESP_LOGE(TAG, "gpio_set_level failed %d, %s", err, esp_err_to_name(err));
223 return err;
224 }
225
226 // This task must be pinned as CPU cycle count is used for profiling
227 int ret = xTaskCreatePinnedToCore(i2s_processing_task, "i2s proc", TASK_STACK, audio_param, TASK_PRIO, NULL, 1);
228 if(ret != pdPASS)
229 {
230 ESP_LOGE(TAG, "Processing task create failed %d", ret);
231 }
232
233 return ESP_OK;
234}
