34template<>
35std::optional<Mapping> from_json(const rapidjson::Value &json)
36{
37 constexpr char TAG[] = "Mapping from_json";
38 auto midi_channel = json.FindMember("midi_channel");
39 if(midi_channel == json.MemberEnd()) {
40 ESP_LOGE(TAG, "midi_channel is missing");
41 return std::nullopt;
42 }
43
44 auto cc_number = json.FindMember("cc_number");
45 if(cc_number == json.MemberEnd()) {
46 ESP_LOGE(TAG, "cc_number is missing");
47 return std::nullopt;
48 }
49
50 Mapping::Mode mode = Mapping::Mode::PARAMETER;
51 auto mode_json = json.FindMember("mode");
52 if(mode_json == json.MemberEnd())
53 {
54 ESP_LOGW(TAG, "mode is missing, setting it to PARAMETER");
55 }
56 else
57 {
58 if(!mode_json->value.IsString())
59 {
60 ESP_LOGE(TAG, "mode is not string");
61 return std::nullopt;
62 }
63
64 static const etl::map<etl::string_view, Mapping::Mode, 2> lut{
65 {"parameter", Mapping::Mode::PARAMETER},
66 {"toggle", Mapping::Mode::TOGGLE},
67 };
68
69 if(!lut.contains(mode_json->value.GetString()))
70 {
71 ESP_LOGE(TAG, "unknown mode %s", mode_json->value.GetString());
72 return std::nullopt;
73 }
74 mode = lut.at(mode_json->value.GetString());
75 }
76
77 auto parameter_id = json.FindMember("parameter_id");
78 if(parameter_id == json.MemberEnd()) {
79 ESP_LOGE(TAG, "parameter_id is missing");
80 return std::nullopt;
81 }
82
83 // TODO range check before narrowing conversion to uint8_t
84 return Mapping{static_cast<uint8_t>(midi_channel->value.GetInt()),
85 static_cast<uint8_t>(cc_number->value.GetInt()),
86 mode,
87 parameters::id_t(parameter_id->value.GetString())};
88}

90template<>
91std::optional<std::pair<Mapping::id_t, Mapping>> from_json(const rapidjson::Value &json)
92{
93 constexpr char TAG[] = "pair<mapping::id, mapping> from_json";
94
95 if(!json.IsObject())
96 {
97 ESP_LOGE(TAG, "mapping is not object");
98 return std::nullopt;
99 }
100
101 // XXX There should be only one key, so we take the first one
102 // and ignore the rest
103 auto mapping_entry_member = json.GetObject().begin();
104
105 if(mapping_entry_member == json.GetObject().end())
106 {
107 ESP_LOGE(TAG, "mapping is empty");
108 return std::nullopt;
109 }
110
111 auto &mapping_id = mapping_entry_member->name;
112 auto &mapping_entry = mapping_entry_member->value;
113
114 auto mapping = from_json<Mapping>(mapping_entry);
115 if(!mapping.has_value()) {
116 ESP_LOGE(TAG, "failed to get mapping");
117 return std::nullopt;
118 }
119
120 auto id = from_json<Mapping::id_t>(mapping_id);
121 if(!id.has_value()) {
122 ESP_LOGE(TAG, "failed to get id");
123 return std::nullopt;
124 }
125
126 return {{ *id, *mapping }};
127}

179template<>
180std::optional<MappingApiMessage> from_json(const rapidjson::Value &json) {
181 constexpr char TAG[] = "MidiApiMessage from_json";
182
183 auto message_type_member = json.FindMember("messageType");
184 if(message_type_member == json.MemberEnd()) {
185 ESP_LOGE(TAG, "messageType is missing");
186 goto error;
187 }
188
189 if(!message_type_member->value.IsString()) {
190 ESP_LOGE(TAG, "messageType is not string");
191 goto error;
192 }
193
194 {
195 const etl::map<std::string, std::function<std::optional<MappingApiMessage>()>, 4> lut{
196 { get_message_type<GetRequest>(), [&]{return from_json<GetRequest>(json);} },
197 { get_message_type<CreateRequest>(), [&]{return from_json<CreateRequest>(json);} },
198 { get_message_type<Update>(), [&]{return from_json<Update>(json);} },
199 { get_message_type<Remove>(), [&]{return from_json<Remove>(json);} },
200 };
201
202 const char *message_type = message_type_member->value.GetString();
203 if(auto f = lut.find(message_type); f != lut.end())
204 {
205 return f->second();
206 }
207 }
208
209error:
210 return std::nullopt;
211}

232static int parse_uuid(Mapping::id_t &uuid, const char *string)
233{
234 constexpr std::size_t UUID_LENGTH = 36;
235 constexpr char TAG[] = "parse_uuid";
236
237 if(UUID_LENGTH != std::strlen(string))
238 {
239 ESP_LOGE(TAG, "Incorrect UUID string length");
240 return -1;
241 }
242
243 size_t i = 0;
244 size_t j = 0;
245 ESP_LOGD(TAG, "i = %zu, j = %zu", i , j);
246 while(i < UUID_LENGTH)
247 {
248 char digit[2];
249 std::memcpy(digit, &string[i], 2);
250
251 ESP_LOGD(TAG, "digit = %c%c", digit[0] , digit[1]);
252
253 // TODO error on invalid characters: z, symbols etc
254 // The only valid characters are 0 to 9 and a to f.
255 //
256 // Handle uppercase as well. It is required when a string UUID
257 // is an input as per RFC 4122 Section 3
258 // https://tools.ietf.org/html/rfc4122#section-3
259 auto get_value = [&] (char hex) -> uint8_t {
260 if(hex >= 'a')
261 {
262 return hex - 'a' + 10;
263 }
264 else
265 {
266 return hex - '0';
267 }
268 };
269
270 uuid[j] = get_value(digit[0]) << 4 | get_value(digit[1]);
271
272 j++;
273
274 i += 2;
275 bool is_separator = (i == 8) || (i == 13) || (i == 18) || (i == 23);
276 if(is_separator) i++;
277 ESP_LOGD(TAG, "i = %zu, j = %zu", i , j);
278 }
279
280 return 0;
281}
