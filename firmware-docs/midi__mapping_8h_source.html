213using MappingObserver = etl::observer<const Mapping::id_t&>;

215template<typename AudioParametersT, std::size_t MAX_MAPPINGS, std::size_t MAX_OBSERVERS>
216class MappingManager final : public etl::observable<MappingObserver, MAX_OBSERVERS> {
217 public:
218 using MapType = etl::map<Mapping::id_t, Mapping, MAX_MAPPINGS>;
219
220 explicit MappingManager(std::shared_ptr<AudioParametersT> a_parameters) : parameters{a_parameters} {}
221 MappingManager(std::shared_ptr<AudioParametersT> a_parameters, MapType initial_mappings) : parameters{a_parameters}, mappings{initial_mappings} {}
222
223 [[nodiscard]] const etl::imap<Mapping::id_t, Mapping> *get() const {
224 return &mappings;
225 }
227 [[nodiscard]] int create(const std::pair<const Mapping::id_t, Mapping> &mapping) {
228 if(mappings.full()) {
229 ESP_LOGE(TAG, "Failed to create new midi mapping, map is full");
230 return -1;
231 }
232
233 mappings.insert(mapping);
234 this->notify_observers(mapping.first);
235 return 0;
236 }
238 [[nodiscard]] int update(const std::pair<const Mapping::id_t, Mapping> &mapping) {
239 if(!mappings.contains(mapping.first))
240 {
241 ESP_LOGE(TAG, "Does not contain key");
242 ESP_LOG_BUFFER_HEX_LEVEL(TAG, mapping.first.data(), mapping.first.size(), ESP_LOG_ERROR);
243 return -1;
244 }
245
246 mappings.erase(mapping.first);
247 mappings.insert(mapping);
248 this->notify_observers(mapping.first);
249 return 0;
250 }
251 void remove(const Mapping::id_t &id) {
252 mappings.erase(id);
253 this->notify_observers(id);
254 }
255
259 void process_message(Message message) const {
260 auto cc_params = get_if<Message::ControlChange>(&message.parameters);
261 if(!cc_params) return;
262
263 for(const auto &[_, mapping] : mappings)
264 {
265 if(mapping.midi_channel != message.channel)
266 {
267 continue;
268 }
269
270 if(mapping.cc_number != cc_params->control)
271 {
272 continue;
273 }
274
275 switch(mapping.mode)
276 {
277 case Mapping::Mode::PARAMETER:
278 parameters->update(mapping.parameter_name,
279 cc_params->value / float(CC_VALUE_MAX));
280 break;
281 case Mapping::Mode::TOGGLE:
282 if(cc_params->value == 0)
283 {
284 continue;
285 }
286
287 auto old_value = parameters->get(mapping.parameter_name);
288
289 parameters->update(mapping.parameter_name,
290 old_value < 0.5f ? 1.f : 0.f);
291 break;
292 }
293 }
294 };
295
296 private:
297 std::shared_ptr<AudioParametersT> parameters;
298 MapType mappings;
299
300 static constexpr char TAG[] = "midi_mapping";
301};
