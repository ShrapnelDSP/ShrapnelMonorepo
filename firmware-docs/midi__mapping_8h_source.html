57using MappingObserver = etl::observer<const Mapping::id_t &>;

59template <typename AudioParametersT,
60 std::size_t MAX_MAPPINGS,
61 std::size_t MAX_OBSERVERS>
62class MappingManager final
63 : public etl::observable<MappingObserver, MAX_OBSERVERS>,
64 public persistence::Crud<Mapping>
65{
66public:
67 using MapType = etl::map<Mapping::id_t, Mapping, MAX_MAPPINGS>;
68
69 MappingManager(
70 std::shared_ptr<AudioParametersT> a_parameters,
71 std::unique_ptr<persistence::Crud<std::span<uint8_t>>> a_storage)
72 : parameters{a_parameters},
73 storage{std::move(a_storage)}
74 {
75 for_each(
76 [this](uint32_t id, const shrapnel::midi::Mapping &mapping) {
77 mappings.insert({id, mapping});
78 });
79 }
80
81 [[nodiscard]] const etl::imap<Mapping::id_t, Mapping> *get() const
82 {
83 return &mappings;
84 }
85
87 [[nodiscard]] int create(const Mapping &mapping,
88 Mapping::id_t &id_out) override
89 {
90 if(mappings.full())
91 {
92 ESP_LOGE(TAG, "Failed to create new midi mapping, map is full");
93 return -1;
94 }
95
96 std::array<uint8_t, 256> memory{};
97 std::span<uint8_t> buffer{memory};
98
99 auto encoded = api::to_bytes(mapping, buffer);
100 if(!encoded.has_value())
101 {
102 return -1;
103 }
104
105 int rc = storage->create(*encoded, id_out);
106 if(rc != 0)
107 {
108 return -1;
109 }
110
111 mappings.insert({id_out, mapping});
112 this->notify_observers(id_out);
113 return 0;
114 }
115
116 [[nodiscard]] int read(Mapping::id_t id, Mapping &mapping) override
117 {
118 std::array<uint8_t, 256> memory{};
119 std::span<uint8_t> buffer{memory};
120
121 int rc = storage->read(id, buffer);
122 if(rc != 0)
123 {
124 return -1;
125 }
126
127 auto decoded = api::from_bytes<Mapping>(buffer);
128 if(!decoded.has_value())
129 {
130 return -1;
131 }
132
133 mappings.insert({id, *decoded});
134 mapping = *decoded;
135 this->notify_observers(id);
136 return 0;
137 }
138
140 [[nodiscard]] int update(Mapping::id_t id, const Mapping &mapping) override
141 {
142 if(!mappings.contains(id))
143 {
144 ESP_LOGE(TAG, "Does not contain key %" PRIu32, id);
145 return -1;
146 }
147
148 std::array<uint8_t, 256> memory{};
149 std::span<uint8_t> buffer{memory};
150
151 auto encoded = api::to_bytes(mapping, buffer);
152 if(!encoded.has_value())
153 {
154 return -1;
155 }
156
157 int rc = storage->update(id, *encoded);
158 if(rc != 0)
159 {
160 return -1;
161 }
162
163 mappings.erase(id);
164 mappings.insert({id, mapping});
165 this->notify_observers(id);
166 return 0;
167 }
168
169 [[nodiscard]] int destroy(Mapping::id_t id) override
170 {
171 int rc = storage->destroy(id);
172 if(rc != 0)
173 {
174 return -1;
175 }
176
177 mappings.erase(id);
178 this->notify_observers(id);
179 return 0;
180 }
181
182 void
183 for_each(etl::delegate<void(uint32_t, const Mapping &)> callback) override
184 {
185 storage->for_each(
186 [&callback](uint32_t id, const std::span<uint8_t> &buffer)
187 {
188 auto decoded = api::from_bytes<Mapping>(buffer);
189 if(!decoded.has_value())
190 {
191 return;
192 }
193
194 callback(id, *decoded);
195 });
196 }
197
201 void process_message(Message message) const
202 {
203 auto cc_params = get_if<Message::ControlChange>(&message.parameters);
204 if(!cc_params)
205 return;
206
207 for(const auto &[_, mapping] : mappings)
208 {
209 if(mapping.midi_channel != message.channel)
210 {
211 continue;
212 }
213
214 if(mapping.cc_number != cc_params->control)
215 {
216 continue;
217 }
218
219 switch(mapping.mode)
220 {
221 case Mapping::Mode::PARAMETER:
222 parameters->update(mapping.parameter_name,
223 cc_params->value / float(CC_VALUE_MAX));
224 break;
225 case Mapping::Mode::TOGGLE:
226 if(cc_params->value == 0)
227 {
228 continue;
229 }
230
231 auto old_value = parameters->get(mapping.parameter_name);
232
233 parameters->update(mapping.parameter_name,
234 old_value < 0.5f ? 1.f : 0.f);
235 break;
236 }
237 }
238 };
239
240private:
241 std::shared_ptr<AudioParametersT> parameters;
242 std::unique_ptr<persistence::Crud<std::span<uint8_t>>> storage;
243 MapType mappings;
244
245 static constexpr char TAG[] = "midi_mapping";
246};
