181struct Mapping {
182 using id_t = std::array<uint8_t, 16>;
183
184 uint8_t midi_channel;

188 std::strong_ordering operator<=>(const Mapping &other) const = default;
189};

191using MappingObserver = etl::observer<const Mapping::id_t&>;
192
193template<typename AudioParametersT, std::size_t MAX_MAPPINGS, std::size_t MAX_OBSERVERS>
194class MappingManager final : public etl::observable<MappingObserver, MAX_OBSERVERS> {
195 public:
196 using MapType = etl::map<Mapping::id_t, Mapping, MAX_MAPPINGS>;
197
198 explicit MappingManager(std::shared_ptr<AudioParametersT> a_parameters) : parameters{a_parameters} {}
199 MappingManager(std::shared_ptr<AudioParametersT> a_parameters, MapType initial_mappings) : parameters{a_parameters}, mappings{initial_mappings} {}
200
201 [[nodiscard]] const etl::imap<Mapping::id_t, Mapping> *get() const {
202 return &mappings;
203 }
205 [[nodiscard]] int create(const std::pair<const Mapping::id_t, Mapping> &mapping) {
206 if(mappings.full()) {
207 ESP_LOGE(TAG, "Failed to create new midi mapping, map is full");
208 return -1;
209 }
210
211 mappings.insert(mapping);
212 this->notify_observers(mapping.first);
213 return 0;
214 }
216 [[nodiscard]] int update(const std::pair<const Mapping::id_t, Mapping> &mapping) {
217 if(!mappings.contains(mapping.first))
218 {
219 ESP_LOGE(TAG, "Does not contain key");
220 ESP_LOG_BUFFER_HEX_LEVEL(TAG, mapping.first.data(), mapping.first.size(), ESP_LOG_ERROR);
221 return -1;
222 }
223
224 mappings.erase(mapping.first);
225 mappings.insert(mapping);
226 this->notify_observers(mapping.first);
227 return 0;
228 }
229 void remove(const Mapping::id_t &id) {
230 mappings.erase(id);
231 this->notify_observers(id);
232 }
233
237 void process_message(Message message) const {
238 auto cc_params = get_if<Message::ControlChange>(&message.parameters);
239 if(!cc_params) return;
240
241 for(const auto &mapping : mappings)
242 {
243 if(mapping.second.midi_channel != message.channel)
244 {
245 continue;
246 }
247
248 if(mapping.second.cc_number != cc_params->control)
249 {
250 continue;
251 }
252
253 parameters->update(
254 mapping.second.parameter_name,
255 cc_params->value / float(CC_VALUE_MAX));
256 }
257 };
258
259 private:
260 std::shared_ptr<AudioParametersT> parameters;
261 MapType mappings;
262
263 static constexpr char TAG[] = "midi_mapping";
264};
