286 [[nodiscard]] const etl::imap<Mapping::id_t, Mapping> *get() const {

290 [[nodiscard]] int create(const std::pair<const Mapping::id_t, Mapping> &mapping) {
291 if(mappings.full()) {
292 ESP_LOGE(TAG, "Failed to create new midi mapping, map is full");
293 return -1;
294 }
295
296 mappings.insert(mapping);
297 this->notify_observers(mapping.first);
298 return 0;
299 }
301 [[nodiscard]] int update(const std::pair<const Mapping::id_t, Mapping> &mapping) {
302 if(!mappings.contains(mapping.first))
303 {
304 ESP_LOGE(TAG, "Does not contain key");
305 ESP_LOG_BUFFER_HEX_LEVEL(TAG, mapping.first.data(), mapping.first.size(), ESP_LOG_ERROR);
306 return -1;
307 }
308
309 mappings.erase(mapping.first);
310 mappings.insert(mapping);
311 this->notify_observers(mapping.first);
312 return 0;
313 }
314 void remove(const Mapping::id_t &id) {
315 mappings.erase(id);
316 this->notify_observers(id);
317 }

322 void process_message(Message message) const {
323 auto cc_params = get_if<Message::ControlChange>(&message.parameters);
324 if(!cc_params) return;
325
326 for(const auto &[_, mapping] : mappings)
327 {
328 if(mapping.midi_channel != message.channel)
329 {
330 continue;
331 }
332
333 if(mapping.cc_number != cc_params->control)
334 {
335 continue;
336 }
337
338 switch(mapping.mode)
339 {
340 case Mapping::Mode::PARAMETER:
341 parameters->update(mapping.parameter_name,
342 cc_params->value / float(CC_VALUE_MAX));
343 break;
344 case Mapping::Mode::TOGGLE:
345 if(cc_params->value == 0)
346 {
347 continue;
348 }
349
350 auto old_value = parameters->get(mapping.parameter_name);
351
352 parameters->update(mapping.parameter_name,
353 old_value < 0.5f ? 1.f : 0.f);
354 break;
355 }
356 }
357 };
358
359 private:
360 std::shared_ptr<AudioParametersT> parameters;
361 MapType mappings;
362
363 static constexpr char TAG[] = "midi_mapping";
364};
