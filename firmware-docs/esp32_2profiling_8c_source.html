32static uint32_t start_cycles = 0;
33static uint32_t stage_cycles[NUMBER_OF_STAGES] = {};
34static SemaphoreHandle_t profiling_mutex;
35static bool got_semaphore = false;
36static uint32_t cpu_freq_mhz;
37static size_t buffer_size;

40void profiling_init(size_t a_buffer_size, float a_sample_rate)
41{
42 profiling_mutex = xSemaphoreCreateMutex();
43 buffer_size = a_buffer_size;
44 sample_rate = a_sample_rate;
45
46 cpu_freq_mhz = ets_get_cpu_frequency();
47 ESP_LOGI(TAG, "CPU frequency: %d MHz", cpu_freq_mhz);
48}
49
50void profiling_start(void)
51{
52 if(xSemaphoreTake(profiling_mutex, 0))
53 {
54 got_semaphore = true;
55 start_cycles = esp_cpu_get_cycle_count();
56 }
57}

59void profiling_mark_stage(unsigned int stage)
60{
61 assert(stage < NUMBER_OF_STAGES);
62
63 if(got_semaphore)
64 {
65 stage_cycles[stage] = esp_cpu_get_cycle_count();
66 }
67}
68
69void profiling_stop(void)
70{
71 if(got_semaphore)
72 {
73 xSemaphoreGive(profiling_mutex);
74 got_semaphore = false;
75 }
76}

78static double cycles_to_percent(int64_t cycles)
79{
80 float cycles_per_second = cpu_freq_mhz * 1e6f;
81
82 float ratio =
83 cycles / ((float)buffer_size / sample_rate) / (cycles_per_second);
84
85 return (double)(100 * ratio);
86}

88static int64_t cycles_to_us(int64_t cycles) { return cycles / cpu_freq_mhz; }

90void profiling_task(void *param)
91{
92 (void)param;
93 while(1)
94 {
95 vTaskDelay(1000 / portTICK_PERIOD_MS);
96
97 if(xSemaphoreTake(profiling_mutex, 100 / portTICK_PERIOD_MS))
98 {
99 ESP_LOGI(TAG, " ========= \n");
100
101 int64_t total_cycles = 0;
102 for(int i = 0; i < NUMBER_OF_STAGES; i++)
103 {
104 //if the stage has not been assigned, stop printing
105 if(stage_cycles[i] == 0)
106 {
107 break;
108 }
109
110 int64_t current_stage_cycles =
111 stage_cycles[i] -
112 ((i == 0) ? start_cycles : stage_cycles[i - 1]);
113 ESP_LOGI(TAG,
114 "Stage %3d processing took %7lld cycles %4lld us "
115 "(%03.1f %%)",
116 i,
117 current_stage_cycles,
118 cycles_to_us(current_stage_cycles),
119 cycles_to_percent(current_stage_cycles));
120
121 total_cycles += current_stage_cycles;
122 }
123
124 ESP_LOGI(
125 TAG,
126 "Total processing took %7lld cycles %4lld us (%03.1f %%)",
127 total_cycles,
128 cycles_to_us(total_cycles),
129 cycles_to_percent(total_cycles));
130
131 ESP_LOGI(TAG, " ========= \n");
132
133 xSemaphoreGive(profiling_mutex);
134 }
135 else
136 {
137 ESP_LOGE(TAG, "%s failed to take semaphore", __FUNCTION__);
138 }
139 }
140}
