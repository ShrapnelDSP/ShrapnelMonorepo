31#define CLIPPING_REPEAT_TICK (1000/portTICK_PERIOD_MS)
32
33static void audio_event_task(void *parameters)
34{
35 (void) parameters;
36
37 TickType_t last_clipping_tick = 0;
38
39 while(1)
40 {
41 EventBits_t bits = xEventGroupWaitBits(g_audio_event_group, 0xFFFFFF, true, false, portMAX_DELAY);
42
43 if(bits & AUDIO_EVENT_OUTPUT_CLIPPED)
44 {
45
46 /* Clipping messages are throttled to not send more data then
47 * needed */
48 if((xTaskGetTickCount() - last_clipping_tick) > CLIPPING_REPEAT_TICK)
49 {
50 audio_event_send_callback(clipping_message, -1);
51
52 ESP_LOGI(TAG, "Sent clipping message");
53 last_clipping_tick = xTaskGetTickCount();
54 }
55 //clear the bit we just checked
56 bits &= (~(EventBits_t)AUDIO_EVENT_OUTPUT_CLIPPED);
57 }
58
59 if(bits)
60 {
61 ESP_LOGE(TAG, "Some bits were not handled %x", bits);
62 }
63 }
64}

66esp_err_t audio_event_init(void)
67{
68 g_audio_event_group = xEventGroupCreate();
69
70 if(g_audio_event_group == NULL)
71 {
72 ESP_LOGE(TAG, "Failed to create event group");
73 return ESP_FAIL;
74 }
75
76 if(pdPASS != xTaskCreate(audio_event_task, "audio event", 2000, NULL, 4, NULL))
77 {
78 ESP_LOGE(TAG, "Failed to create task");
79 return ESP_FAIL;
80 }
81
82 return ESP_OK;
83}
