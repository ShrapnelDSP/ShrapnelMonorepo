33void get_device_service_name(char *service_name, size_t max)
34{
35 uint8_t eth_mac[6];
36 const char *ssid_prefix = "shrapnel_";
37 esp_wifi_get_mac(WIFI_IF_STA, eth_mac);
38 snprintf(service_name,
39 max,
40 "%s%02X%02X%02X",
41 ssid_prefix,
42 eth_mac[3],
43 eth_mac[4],
44 eth_mac[5]);
45}

47void provisioning_event_handler(void* arg, esp_event_base_t event_base,
48 int32_t event_id, void* event_data)
49{
50 assert(arg != nullptr);
51 assert(event_base == WIFI_PROV_EVENT);
52
53 auto send_event = static_cast<
54 shrapnel::wifi::WifiStateMachine::internal_event_callback_t *>(arg);
55
56 if (event_base == WIFI_PROV_EVENT) {
57 switch (event_id) {
58 case WIFI_PROV_INIT:
59 ESP_LOGI(TAG, "Provisioning initialised");
60 break;
61 case WIFI_PROV_START:
62 ESP_LOGI(TAG, "Provisioning started");
63 break;
64 case WIFI_PROV_CRED_RECV: {
65 wifi_sta_config_t *wifi_sta_cfg = (wifi_sta_config_t *)event_data;
66 ESP_LOGI(TAG, "Received Wi-Fi credentials");
67 ESP_LOGI(TAG, "SSID:");
68 ESP_LOG_BUFFER_HEXDUMP(
69 TAG,
70 wifi_sta_cfg->ssid,
71 sizeof(wifi_sta_cfg->ssid),
72 ESP_LOG_INFO);
73 ESP_LOGI(TAG, "Password:");
74 ESP_LOG_BUFFER_HEXDUMP( TAG,
75 wifi_sta_cfg->password,
76 sizeof(wifi_sta_cfg->password),
77 ESP_LOG_INFO);
78 break;
79 }
80 case WIFI_PROV_CRED_FAIL: {
81 auto reason = (wifi_prov_sta_fail_reason_t *)event_data;
82 ESP_LOGE(TAG, "Provisioning failed!\n\tReason : %s",
83 (*reason == WIFI_PROV_STA_AUTH_ERROR) ?
84 "Wi-Fi station authentication failed" : "Wi-Fi access-point not found");
85
86 (*send_event)(shrapnel::wifi::InternalEvent::PROVISIONING_FAILURE);
87 break;
88 }
89 case WIFI_PROV_CRED_SUCCESS: {
90 ESP_LOGI(TAG, "Provisioning successful");
91
92 (*send_event)(shrapnel::wifi::InternalEvent::PROVISIONING_SUCCESS);
93 break;
94 }
95 case WIFI_PROV_END:
96 ESP_LOGI(TAG, "Provisioning end");
97 (*send_event)(shrapnel::wifi::InternalEvent::PROVISIONING_END);
98 break;
99 default:
100 ESP_LOGW(TAG, "Unhandled provisioning event: %d", event_id);
101 break;
102 }
103 }
104}

106void wifi_event_handler(void* arg, esp_event_base_t event_base,
107 int32_t event_id, void* event_data)
108{
109 (void) arg;
110
111 if (event_base == WIFI_EVENT && event_id == WIFI_EVENT_SCAN_DONE) {
112 ESP_LOGI(TAG, "WIFI_EVENT_SCAN_DONE");
113 } else if (event_base == WIFI_EVENT && event_id == WIFI_EVENT_STA_START) {
114 ESP_LOGI(TAG, "WIFI_EVENT_STA_START");
115 } else if (event_base == WIFI_EVENT && event_id == WIFI_EVENT_STA_CONNECTED) {
116 ESP_LOGI(TAG, "WIFI_EVENT_STA_CONNECTED");
117 } else if (event_base == WIFI_EVENT && event_id == WIFI_EVENT_STA_DISCONNECTED) {
118 ESP_LOGE(TAG, "WIFI_EVENT_STA_DISCONNECTED");
119 } else if (event_base == IP_EVENT && event_id == IP_EVENT_STA_GOT_IP) {
120 ip_event_got_ip_t* event = (ip_event_got_ip_t*) event_data;
121 ESP_LOGI(TAG, "Connected with IP Address:" IPSTR, IP2STR(&event->ip_info.ip));
122 } else if (event_base == WIFI_EVENT && event_id == WIFI_EVENT_AP_START) {
123 ESP_LOGI(TAG, "WIFI_EVENT_AP_START");
124 } else if (event_base == WIFI_EVENT && event_id == WIFI_EVENT_AP_STOP) {
125 ESP_LOGI(TAG, "WIFI_EVENT_AP_STOP");
126 } else if (event_base == WIFI_EVENT && event_id == WIFI_EVENT_AP_STACONNECTED) {
127 ESP_LOGI(TAG, "WIFI_EVENT_AP_STACONNECTED");
128 } else if (event_base == WIFI_EVENT && event_id == WIFI_EVENT_AP_STADISCONNECTED) {
129 ESP_LOGE(TAG, "WIFI_EVENT_AP_STADISCONNECTED");
130 } else {
131 ESP_LOGW(TAG, "Unhandled wifi event: %s %d", event_base, event_id);
132 }
133}

135bool is_provisioned(void)
136{
137 bool provisioned = false;
138
139 wifi_config_t wifi_cfg;
140 ESP_ERROR_CHECK(esp_wifi_get_config(WIFI_IF_STA, &wifi_cfg));
141
142 if (wifi_cfg.sta.ssid[0] != '\0') {
143 provisioned = true;
144 }
145 return provisioned;
146}

161constexpr WifiStateMachine::transition WifiStateMachine::transition_table[]{
162 // clang-format off
163 transition(State::INIT, InternalEvent::IS_PROVISIONED, State::STARTING),
164 transition(State::INIT, InternalEvent::IS_NOT_PROVISIONED, State::PROVISIONING, &WifiStateMachine::provisioning_init),
165
166 transition(State::STARTING, InternalEvent::STARTED, State::CONNECTING),
167
168 // Espressif uses the disconnect event to indicate failed connection. Retry
169 // when we receive this event, but only if it has been less than
170 // CONNECT_TIMEOUT_MS since we have started connecting. Otherwise, return
171 // to the provisioning state.
172 transition(State::CONNECTING, InternalEvent::DISCONNECT, State::CONNECTING, &WifiStateMachine::connect_to_ap, &WifiStateMachine::is_reconnecting),
173 transition(State::CONNECTING, InternalEvent::DISCONNECT, State::PROVISIONING, &WifiStateMachine::provisioning_init),
174 transition(State::CONNECTING, InternalEvent::CONNECT_SUCCESS, State::CONNECTED),
175
176 transition(State::CONNECTED, InternalEvent::DISCONNECT, State::CONNECTING),
177 transition(State::CONNECTED, InternalEvent::RESET_PROVISIONING, State::PROVISIONING, &WifiStateMachine::provisioning_init),
178
179 transition(State::PROVISIONING, InternalEvent::PROVISIONING_SUCCESS, State::PROVISIONING_FINISHING),
180 transition(State::PROVISIONING, InternalEvent::PROVISIONING_FAILURE, State::PROVISIONING_STOPPING),
181
182 transition(State::PROVISIONING_FINISHING, InternalEvent::PROVISIONING_END, State::CONNECTED, &WifiStateMachine::provisioning_deinit),
183
184 transition(State::PROVISIONING_STOPPING, InternalEvent::PROVISIONING_END, State::PROVISIONING),
185 // clang-format on
186};

200void WifiStateMachine::check_if_provisioned()
201{
202 bool is_provisioned = ::is_provisioned();
203#if SHRAPNEL_RESET_WIFI_CREDENTIALS
204 ESP_LOGW(TAG, "Reseting wifi provisioning");
205 is_provisioned = false;
206#endif
207 send_event_internal(is_provisioned ? InternalEvent::IS_PROVISIONED : InternalEvent::IS_NOT_PROVISIONED);
208}

232bool WifiStateMachine::is_reconnecting()
233{
234 bool is_reconnecting = (xTaskGetTickCount() - connect_start_tick_count) < CONNECT_TIMEOUT_MS;
235 if(is_reconnecting)
236 {
237 ESP_LOGI(TAG, "Retrying connection");
238 }
239 else
240 {
241 ESP_LOGI(TAG, "Connection timed out, giving up");
242 }
243 return is_reconnecting;
244}

258void WifiStateMachine::provisioning_init()
259{
260 // XXX: Make sure to deregister the handlers in the destructor
261 ESP_ERROR_CHECK(esp_event_handler_register(
262 WIFI_PROV_EVENT,
263 ESP_EVENT_ANY_ID,
264 provisioning_event_handler,
265 &send_event_internal));
266 ESP_ERROR_CHECK(esp_event_handler_register(
267 WIFI_EVENT,
268 ESP_EVENT_ANY_ID,
269 wifi_event_handler,
270 nullptr));
271 ESP_ERROR_CHECK(esp_event_handler_register(
272 IP_EVENT,
273 IP_EVENT_STA_GOT_IP,
274 wifi_event_handler,
275 nullptr));
276
277 wifi_prov_mgr_config_t config = {
278 .scheme{wifi_prov_scheme_softap},
279 .scheme_event_handler{.event_cb{nullptr}, .user_data{nullptr}},
280 .app_event_handler{.event_cb{nullptr}, .user_data{nullptr}},
281 };
282
283 ESP_ERROR_CHECK(wifi_prov_mgr_init(config));
284}
