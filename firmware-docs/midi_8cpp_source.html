33void Decoder::decode(uint8_t byte)
34{
35 State new_state = IDLE;
36 switch(state)
37 {
38 case IDLE:
39 new_state = decode_idle(byte);
40 break;
41 case GOT_MESSAGE:
42 new_state = decode_message(byte);
43 break;
44 default:
45 ESP_LOGE(TAG, "Unhandled decoder state %d", state);
46 break;
47 }
48
49 state = new_state;
50}

52Decoder::State Decoder::decode_idle(uint8_t byte)
53{
54 /* Running status allows data to be sent with no status byte. In this case,
55 * the previous status is reused. */
56 if(!is_status_byte(byte) && is_status_byte(current_status))
57 {
58 auto next_state = decode_message(byte);
59 return next_state;
60 }
61
62 switch(byte & 0xF0)
63 {
64 case CONTROL_CHANGE:
65 case NOTE_ON:
66 case NOTE_OFF:
67 case PROGRAM_CHANGE:
68 current_status = byte;
69 return GOT_MESSAGE;
70 }
71 return IDLE;
72}

74Decoder::State Decoder::decode_message(uint8_t byte)
75{
76 assert(is_status_byte(current_status));
77
78 switch(current_status & 0xF0)
79 {
80 case CONTROL_CHANGE:
81 case NOTE_ON:
82 case NOTE_OFF:
83 received_data[data_count] = byte;
84 data_count++;
85 if(data_count == 2)
86 {
87 output_message();
88 data_count = 0;
89 return IDLE;
90 }
91 break;
92 case PROGRAM_CHANGE:
93 received_data[0] = byte;
94 output_message();
95 return IDLE;
96 default:
97 assert(false);
98 break;
99 }
100
101 return GOT_MESSAGE;
102}

104void Decoder::output_message()
105{
106 Message message{};
107
108 switch(current_status & 0xF0)
109 {
110 case NOTE_ON:
111 // TODO how to use a designated initialiser here? This is pretty hard
112 // to read.
113 message.parameters = Message::NoteOn{
114 .note = received_data[0],
115 .velocity = received_data[1]};
116 break;
117 case NOTE_OFF:
118 message.parameters = Message::NoteOff{
119 .note = received_data[0],
120 .velocity = received_data[1]};
121 break;
122 case CONTROL_CHANGE:
123 // Channel Mode messages should be ignored
124 if((received_data[0] & 0xF8) == 0x78)
125 {
126 return;
127 }
128
129 message.parameters = Message::ControlChange{
130 .control = received_data[0],
131 .value = received_data[1]};
132 break;
133 case PROGRAM_CHANGE:
134 message.parameters = Message::ProgramChange{.number = received_data[0]};
135 break;
136 default:
137 assert(false);
138 break;
139 }
140
141 message.channel = (current_status & 0x0F) + 1;
142 on_message_decoded(message);
143}
