34template<std::size_t N, std::size_t K>
35class FastConvolution final {
36 public:
37 FastConvolution(const std::array<float, K> &a)
38 {
39 ESP_ERROR_CHECK(dsps_fft4r_init_fc32(NULL, CONFIG_DSP_MAX_FFT_SIZE));
40
41 // transform a
42 std::array<float, N> a_copy{};
43 std::copy(a.cbegin(), a.cend(), a_copy.begin());
44 a_complex = real_to_complex(a_copy);
45 ESP_ERROR_CHECK(dsps_fft4r_fc32(reinterpret_cast<float *>(a_complex.data()), N));
46 dsps_bit_rev4r_fc32(reinterpret_cast<float *>(a_complex.data()), N);
47 }
48
54 void process(
55 const std::array<float, N> &b,
56 std::array<float, N> &out)
57 {
58 // transform b
59 auto b_complex = real_to_complex(b);
60 profiling_mark_stage(7);
61 ESP_ERROR_CHECK(dsps_fft4r_fc32(reinterpret_cast<float *>(b_complex.data()), N));
62 profiling_mark_stage(8);
63 dsps_bit_rev4r_fc32(reinterpret_cast<float *>(b_complex.data()), N);
64 profiling_mark_stage(9);
65
66 // multiply A * B
67 std::array<std::complex<float>, N> multiplied;
68 complex_multiply(a_complex, b_complex, multiplied);
69 profiling_mark_stage(10);
70
71 // transform result
72 // Inverse transform achieved by complex conjugating the input and
73 // output of the forward transform. The imaginary part of the output is
74 // not used, so we don't actually calculate the complex conjugate on
75 // it.
76 //
77 // There is no inverse transform provided by esp-dsp.
78 auto multiplied_ptr = reinterpret_cast<float *>(multiplied.data());
79 dsps_mulc_f32(multiplied_ptr + 1, multiplied_ptr + 1, N, -1, 2, 2);
80 profiling_mark_stage(11);
81 ESP_ERROR_CHECK(dsps_fft4r_fc32(multiplied_ptr, N));
82 profiling_mark_stage(12);
83 dsps_bit_rev4r_fc32(multiplied_ptr, N);
84 profiling_mark_stage(13);
85
86 complex_to_real(multiplied, out);
87 profiling_mark_stage(14);
88
89 dsps_mulc_f32(out.data(), out.data(), N, scale_factor, 1, 1);
90 profiling_mark_stage(15);
91 }
92
93 private:
94 float scale_factor = 1.f/N;
95 std::array<std::complex<float>, N> a_complex;
96
97 std::array<std::complex<float>, N> real_to_complex(const std::array<float, N> &real)
98 {
99 std::array<std::complex<float>, N> _complex{};
100
101 for(std::size_t i = 0; i < N; i++)
102 {
103 _complex[i] = real[i];
104 }
105
106 return _complex;
107 }
108
109 void complex_to_real(
110 const std::array<std::complex<float>, N> &_complex,
111 std::array<float, N> &real)
112 {
113 for(std::size_t i = 0; i < N; i++)
114 {
115 real[i] = _complex[i].real();
116 }
117 }
118
119 public:
120 static void complex_multiply(
121 const std::array<std::complex<float>, N> &a,
122 const std::array<std::complex<float>, N> &b,
123 std::array<std::complex<float>, N> &out)
124 {
125 // We want to compute (r_a + im_a j) * (r_b + im_b j)
126 // This can be rewritten as follows:
127 //
128 // ra rb + ra im_b j + im_a r_b j + im_a im_b j j
129 //
130 // j j is -1 so:
131 //
132 // (ra rb + ra im_b j) + // part 1
133 // (- im_a im_b + im_a r_b j) // part 2
134
135 // TODO we multiply then add here, could we use the MADD.S instruction
136 // to speed it up?
137
138 auto a_ptr = reinterpret_cast<const float*>(a.data());
139 auto b_ptr = reinterpret_cast<const float*>(b.data());
140 auto out_ptr = reinterpret_cast<float*>(out.data());
141 // part 1
142 dsps_mul_f32(a_ptr, b_ptr, out_ptr, N, 2, 2, 2);
143 dsps_mul_f32(a_ptr, b_ptr + 1, out_ptr + 1, N, 2, 2, 2);
144
145 // part 2
146 std::array<std::complex<float>, N> out2;
147 auto out2_ptr = reinterpret_cast<float*>(out2.data());
148 dsps_mul_f32(a_ptr + 1, b_ptr + 1, out2_ptr, N, 2, 2, 2);
149 dsps_mul_f32(a_ptr + 1, b_ptr, out2_ptr + 1, N, 2, 2, 2);
150 dsps_mulc_f32(out2_ptr, out2_ptr, N, -1, 2, 2);
151
152 dsps_add_f32(out_ptr, out2_ptr, out_ptr, 2*out.size(), 1, 1, 1);
153 }
154};
