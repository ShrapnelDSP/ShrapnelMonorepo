102static esp_err_t websocket_get_handler(httpd_req_t *req)
103{
104 CommandHandling<parameters::AudioParameters>::Message message{};
105
106 httpd_ws_frame_t pkt = {
107 .final = false,
108 .fragmented = false,
109 .type = HTTPD_WS_TYPE_TEXT,
110 .payload = reinterpret_cast<uint8_t*>(message.json),
111 .len = 0,
112 };
113
114 if(req->method == HTTP_GET)
115 {
116 ESP_LOGI(TAG, "Got websocket upgrade request");
117 return ESP_OK;
118 }
119
120 esp_err_t rc = httpd_ws_recv_frame(req, &pkt, sizeof(message.json));
121 if(rc != ESP_OK)
122 {
123 ESP_LOGE(TAG, "websocket parse failed");
124 return rc;
125 }
126
127 /* We should never see any of these packets */
128 assert(pkt.type != HTTPD_WS_TYPE_CONTINUE);
129 assert(pkt.type != HTTPD_WS_TYPE_BINARY);
130 assert(pkt.type != HTTPD_WS_TYPE_CLOSE);
131 assert(pkt.type != HTTPD_WS_TYPE_PING);
132 assert(pkt.type != HTTPD_WS_TYPE_PONG);
133 assert(pkt.final);
134 assert(!pkt.fragmented);
135 assert(pkt.len <= sizeof(message.json));
136
137 ESP_LOGD(TAG, "%s len = %zd", __FUNCTION__, pkt.len);
138 ESP_LOG_BUFFER_HEXDUMP(TAG, message.json, sizeof(message.json), ESP_LOG_VERBOSE);
139
140 message.fd = httpd_req_to_sockfd(req);
141
142 rc = in_queue->send(&message, 100 / portTICK_PERIOD_MS);
143 if(rc != pdTRUE)
144 {
145 ESP_LOGE(TAG, "%s failed to send to queue", __FUNCTION__);
146 }
147
148 rapidjson::Document document;
149 document.Parse(message.json);
150 if(!document.HasParseError())
151 {
152 auto message = midi::from_json<midi::MappingApiMessage>(document.GetObject());
153 if(message.has_value())
154 {
155 etl::string<256> buffer;
156 etl::string_stream stream{buffer};
157 stream << *message;
158 ESP_LOGI(TAG, "decoded %s", buffer.data());
159 }
160 }
161
162 ESP_LOGI(TAG, "%s stack %d", __FUNCTION__, uxTaskGetStackHighWaterMark(NULL));
163
164 return ESP_OK;
165}

167static esp_err_t ui_get_handler(httpd_req_t *req)
168{
169 extern const unsigned char html_start[] asm("_binary_index_html_start");
170 extern const unsigned char html_end[] asm("_binary_index_html_end");
171 const int html_size = (html_end - html_start);
172
173 httpd_resp_send(req, (const char *)html_start, html_size);
174 httpd_resp_send(req, nullptr, 0);
175
176 return ESP_OK;
177}

199static httpd_handle_t start_webserver(void)
200{
201 httpd_handle_t server = NULL;
202 httpd_config_t config = HTTPD_DEFAULT_CONFIG();
203 config.server_port = 8080;
204 config.ctrl_port = 8081;
205 config.max_open_sockets = MAX_CLIENTS;
206
207 // Start the httpd server
208 ESP_LOGI(TAG, "Starting server on port: '%d'", config.server_port);
209 if (httpd_start(&server, &config) == ESP_OK) {
210 // Set URI handlers
211 ESP_LOGI(TAG, "Registering URI handlers");
212 httpd_register_uri_handler(server, &websocket);
213 httpd_register_uri_handler(server, &ui);
214 return server;
215 }
216
217 ESP_LOGE(TAG, "Error starting server!");
218 return NULL;
219}

273typedef struct {
274 char message[MAX_WEBSOCKET_PAYLOAD_SIZE];
275 int fd;
276} audio_event_message_t;

278static void websocket_send(void *arg)
279{
280 httpd_handle_t server = arg;
281
282 audio_event_message_t event_message;
283
284 int rc = xQueueReceive(out_queue, &event_message, 0);
285 if(!rc)
286 {
287 ESP_LOGE(TAG, "%s failed to receive from queue", __FUNCTION__);
288 return;
289 }
290
291 ESP_LOGD(TAG, "%s %d messages waiting", __FUNCTION__, uxQueueMessagesWaiting(out_queue));
292
293 ESP_LOGD(TAG, "%s source fd = %d", __FUNCTION__, event_message.fd);
294
295 ESP_LOGD(TAG, "%s len = %zd", __FUNCTION__, strlen(event_message.message));
296 ESP_LOG_BUFFER_HEXDUMP(TAG, event_message.message, sizeof(event_message.message), ESP_LOG_VERBOSE);
297
298 assert(strlen(event_message.message) < sizeof(event_message.message));
299
300 httpd_ws_frame_t pkt = {
301 .final = false,
302 .fragmented = false,
303 .type = HTTPD_WS_TYPE_TEXT,
304 .payload = reinterpret_cast<uint8_t *>(event_message.message),
305 .len = strlen(event_message.message),
306 };
307
308 int client_fds[MAX_CLIENTS];
309 size_t number_of_clients = ARRAY_LENGTH(client_fds);
310
311 httpd_get_client_list((httpd_handle_t) arg, &number_of_clients, client_fds);
312
313 ESP_LOGD(TAG, "n = %zd", number_of_clients);
314
315 assert(number_of_clients <= MAX_CLIENTS);
316
317 for(size_t i = 0; i < number_of_clients; i++)
318 {
319 int fd = client_fds[i];
320
321 ESP_LOGD(TAG, "fd = %d", fd);
322
323 if(HTTPD_WS_CLIENT_WEBSOCKET != httpd_ws_get_fd_info(server, fd))
324 {
325 continue;
326 }
327
328 if(fd == event_message.fd)
329 {
330 continue;
331 }
332
333 rc = httpd_ws_send_frame_async(server, fd, &pkt);
334 if(ESP_OK != rc)
335 {
336 ESP_LOGE(TAG, "failed to send to fd %d rc %d", fd, rc);
337 }
338 }
339
340 xSemaphoreGive(work_semaphore);
341}

343static void start_mdns(void)
344{
345 esp_err_t err = mdns_init();
346 if (err) {
347 ESP_LOGE(TAG, "MDNS Init failed: %d %s", err, esp_err_to_name(err));
348 return;
349 }
350
351 err = mdns_hostname_set("guitar-dsp");
352 if(err) {
353 ESP_LOGE(TAG, "MDNS failed to set host name %d %s", err, esp_err_to_name(err));
354 return;
355 }
356 err = mdns_instance_name_set("Barabas' Guitar Processor");
357 if(err) {
358 ESP_LOGE(TAG, "MDNS failed to set instance name %d %s", err, esp_err_to_name(err));
359 return;
360 }
361
362 err = mdns_service_add(NULL, "_http", "_tcp", 80, NULL, 0);
363 if(err) {
364 ESP_LOGE(TAG, "MDNS failed to add service %d %s", err, esp_err_to_name(err));
365 return;
366 }
367 err = mdns_service_instance_name_set("_http", "_tcp", "Barabas' Guitar Processor Web Server");
368 if(err) {
369 ESP_LOGE(TAG, "MDNS failed to set service name %d %s", err, esp_err_to_name(err));
370 return;
371 }

399extern "C" void app_main(void)
400{
401 ESP_ERROR_CHECK(heap_caps_register_failed_alloc_callback(failed_alloc_callback));
402
403 ESP_ERROR_CHECK(nvs_flash_init());
404 ESP_ERROR_CHECK(esp_netif_init());
405 ESP_ERROR_CHECK(esp_event_loop_create_default());
406
407 work_semaphore = xSemaphoreCreateBinary();
408 assert(work_semaphore);
409 xSemaphoreGive(work_semaphore);
410
411 in_queue = new Queue<CommandHandling<parameters::AudioParameters>::Message>(QUEUE_LEN);
412 assert(in_queue);
413
414 out_queue = xQueueCreate(QUEUE_LEN, sizeof(audio_event_message_t));
415 assert(out_queue);
416
417 audio_params = new parameters::AudioParameters();
418
419 audio_params->create_and_add_parameter("ampGain", 0, 1, 0.5);
420 audio_params->create_and_add_parameter("ampChannel", 0, 1, 0);
421 audio_params->create_and_add_parameter("bass", 0, 1, 0.5);
422 audio_params->create_and_add_parameter("middle", 0, 1, 0.5);
423 audio_params->create_and_add_parameter("treble", 0, 1, 0.5);
424 //contour gets unstable when set to 0
425 audio_params->create_and_add_parameter("contour", 0.01, 1, 0.5);
426 audio_params->create_and_add_parameter("volume", -30, 0, -15);
427
428 audio_params->create_and_add_parameter("noiseGateThreshold", -80, 0, -60);
429 audio_params->create_and_add_parameter("noiseGateHysteresis", 0, 5, 0);
430 audio_params->create_and_add_parameter("noiseGateAttack", 1, 50, 10);
431 audio_params->create_and_add_parameter("noiseGateHold", 1, 250, 50);
432 audio_params->create_and_add_parameter("noiseGateRelease", 1, 250, 50);
433 audio_params->create_and_add_parameter("noiseGateBypass", 0, 1, 0);
434
435 audio_params->create_and_add_parameter("chorusRate", 0.1, 4, 0.95);
436 audio_params->create_and_add_parameter("chorusDepth", 0, 1, 0.3);
437 audio_params->create_and_add_parameter("chorusMix", 0, 1, 0.8);
438 audio_params->create_and_add_parameter("chorusBypass", 0, 1, 0);
439
440 cmd_handling_task = new CommandHandlingTask<parameters::AudioParameters>(
441 5,
442 in_queue,
443 audio_params,
444 event_send);
445
446 ESP_ERROR_CHECK(audio_event_init());
447
448 i2c_setup();
449
450 profiling_mutex = xSemaphoreCreateMutex();
451 i2s_setup(PROFILING_GPIO, audio_params);
452
453 //dac must be powered up after the i2s clocks have stabilised
454 vTaskDelay(100 / portTICK_PERIOD_MS);
455
456#if HARDWARE == HW_PCB_REV1
457 gpio_config_t io_conf = {
458 .pin_bit_mask = 1ULL << GPIO_CODEC_NRESET,
459 .mode = GPIO_MODE_OUTPUT,
460 .pull_up_en = GPIO_PULLUP_DISABLE,
461 .pull_down_en = GPIO_PULLDOWN_DISABLE,
462 .intr_type = GPIO_INTR_DISABLE,
463 };
464
465 esp_err_t rc = gpio_set_level(GPIO_CODEC_NRESET, 0);
466 if(rc != ESP_OK)
467 {
468 ESP_LOGE(TAG, "gpio_set_level failed %d, %s", rc, esp_err_to_name(rc));
469 }
470
471 rc = gpio_config(&io_conf);
472 if(rc != ESP_OK)
473 {
474 ESP_LOGE(TAG, "gpio_config failed %d, %s", rc, esp_err_to_name(rc));
475 }
476
477 vTaskDelay(5 / portTICK_PERIOD_MS);
478
479 rc = gpio_set_level(GPIO_CODEC_NRESET, 1);
480 if(rc != ESP_OK)
481 {
482 ESP_LOGE(TAG, "gpio_set_level failed %d, %s", rc, esp_err_to_name(rc));
483 }
484
485 vTaskDelay(5 / portTICK_PERIOD_MS);
486
487#endif
488
489 pcm3060_init(I2C_NUM, 0);
490 rc = pcm3060_power_up();
491 if(rc != ESP_OK)
492 {
493 ESP_LOGE(TAG, "pcm3060 power up failed");
494 }
495
496 /* Start the mdns service */
497 start_mdns();
498
499#if 0
500 rc = xTaskCreate(i2s_profiling_task, "i2s profiling", 2000, NULL, tskIDLE_PRIORITY + 2, NULL);
501 if(rc != pdPASS)
502 {
503 ESP_LOGE(TAG, "Profiling task create failed %d", rc);
504 }
505#endif
506
507 esp_netif_create_default_wifi_sta();
508 esp_netif_create_default_wifi_ap();
509 wifi_init_config_t cfg = WIFI_INIT_CONFIG_DEFAULT();
510 ESP_ERROR_CHECK(esp_wifi_init(&cfg));
511
512 /* Start the mdns service */
513 start_mdns();
514
515 {
516 wifi_provisioning::WiFiProvisioning wifi_provisioning{};
517
518#if SHRAPNEL_RESET_WIFI_CREDENTIALS
519 ESP_LOGW(TAG, "Reseting wifi provisioning");
520 wifi_prov_mgr_reset_provisioning();
521#endif
522
523 if(!wifi_provisioning.is_provisioned())
524 {
525 wifi_provisioning.wait_for_provisioning();
526 /* start the websocket server */
527 connect_handler(&_server, IP_EVENT, IP_EVENT_STA_GOT_IP, nullptr);
528 }
529 }
530
531 /* TODO Need to enter provisioning mode again if the WiFi credentials are
532 * not working. Probably the AP passphrase was changed since we got
533 * provisionined.
534 */
535
536 /* Register event handlers to stop the server when Wi-Fi or Ethernet is disconnected,
537 * and re-start it upon connection.
538 */
539 ESP_ERROR_CHECK(esp_event_handler_register(
540 IP_EVENT,
541 IP_EVENT_STA_GOT_IP,
542 connect_handler,
543 &_server));
544 ESP_ERROR_CHECK(esp_event_handler_register(
545 WIFI_EVENT,
546 WIFI_EVENT_STA_DISCONNECTED,
547 disconnect_handler,
548 &_server));
549 ESP_ERROR_CHECK(esp_event_handler_register(
550 WIFI_EVENT,
551 WIFI_EVENT_STA_START,
552 wifi_start_handler,
553 nullptr));
554
555 /* Start Wi-Fi station */
556 ESP_ERROR_CHECK(esp_wifi_set_mode(WIFI_MODE_STA));
557 ESP_ERROR_CHECK(esp_wifi_start());
558
559 ESP_LOGI(TAG, "setup done");
560 ESP_LOGI(TAG, "stack: %d", uxTaskGetStackHighWaterMark(NULL));
561
562 auto midi_uart = new midi::EspMidiUart(UART_NUM_MIDI, GPIO_NUM_MIDI);
563
564 auto log_midi_message = [] (midi::Message message) {
565 etl::string<40> buffer;
566 etl::string_stream stream{buffer};
567 stream << message;
568 ESP_LOGI(TAG, "%s", buffer.data());
569 };
570
571 auto midi_decoder = new midi::Decoder(log_midi_message);
572
573 while(1)
574 {
575
576 uint8_t byte = midi_uart->get_byte();
577 ESP_LOGI(TAG, "midi got byte 0x%02x", byte);
578
579 midi_decoder->decode(byte);
580 }
581}

585extern "C" void audio_event_send_callback(const char *message, int fd)
586{
587 shrapnel::audio_event_message_t event_message = {
588 .message = {0},
589 .fd = fd,
590 };
591
592 snprintf(event_message.message, sizeof(event_message.message), "%s", message);
593
594 ESP_LOGD(TAG, "%s %s", __FUNCTION__, event_message.message);
595 ESP_LOGD(TAG, "%s %s", __FUNCTION__, pcTaskGetName(NULL));
596
597 if(errQUEUE_FULL ==
598 xQueueSendToBack(shrapnel::out_queue,
599 &event_message,
600 100 / portTICK_PERIOD_MS))
601 {
602 ESP_LOGE(TAG, "Failed to send message to websocket");
603 return;
604 }
605
606 if(!xSemaphoreTake(shrapnel::work_semaphore, 1000 / portTICK_PERIOD_MS))
607 {
608 ESP_LOGE(TAG, "work semaphore timed out");
609 return;
610 }
611
612 esp_err_t rc = httpd_queue_work(
613 shrapnel::_server,
614 shrapnel::websocket_send,
615 shrapnel::_server);
616 if(ESP_OK != rc)
617 {
618 ESP_LOGE(TAG, "failed to queue work for server");
619 xQueueReset(shrapnel::out_queue);
620 }
