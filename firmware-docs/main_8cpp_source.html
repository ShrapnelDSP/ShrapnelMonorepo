102static void disconnect_handler(void *arg,
103 esp_event_base_t event_base,
104 int32_t event_id,
105 void *event_data)
106{
107 (void)event_base;
108 (void)event_id;

110
111 ESP_LOGI(TAG, "WiFi disconnected");
112 auto queue{reinterpret_cast<WifiQueue *>(arg)};
113 wifi::InternalEvent event{wifi::InternalEvent::DISCONNECT};
114 int rc{queue->send(&event, 0)};

158static void start_mdns(void)
159{
160 esp_err_t err = mdns_init();
161 if(err)
162 {
163 ESP_LOGE(TAG, "MDNS Init failed: %d %s", err, esp_err_to_name(err));
164 return;
165 }
166
167 err = mdns_hostname_set("guitar-dsp");
168 if(err)
169 {
170 ESP_LOGE(TAG,
171 "MDNS failed to set host name %d %s",
172 err,
173 esp_err_to_name(err));
174 return;
175 }
176 err = mdns_instance_name_set("Barabas' Guitar Processor");
177 if(err)
178 {
179 ESP_LOGE(TAG,
180 "MDNS failed to set instance name %d %s",
181 err,
182 esp_err_to_name(err));
183 return;
184 }
185
186 err = mdns_service_add(NULL, "_http", "_tcp", 80, NULL, 0);
187 if(err)
188 {
189 ESP_LOGE(
190 TAG, "MDNS failed to add service %d %s", err, esp_err_to_name(err));
191 return;
192 }
193 err = mdns_service_instance_name_set(
194 "_http", "_tcp", "Barabas' Guitar Processor Web Server");
195 if(err)
196 {
197 ESP_LOGE(TAG,
198 "MDNS failed to set service name %d %s",
199 err,
200 esp_err_to_name(err));
201 return;
202 }

238extern "C" void app_main(void)
239{
240 ESP_ERROR_CHECK(
241 heap_caps_register_failed_alloc_callback(failed_alloc_callback));
242
243 ESP_ERROR_CHECK(nvs_flash_init());
244 nvs_debug_print();
245 ESP_ERROR_CHECK(esp_netif_init());
246 ESP_ERROR_CHECK(esp_event_loop_create_default());
247
248 auto persistence = std::make_shared<persistence::EspStorage>();
249 auto audio_params = std::make_shared<AudioParameters>();
250
251 i2c_setup();
252 profiling_init(DMA_BUF_SIZE, SAMPLE_RATE);
253
254 //dac must be powered up after the i2s clocks have stabilised
255 vTaskDelay(100 / portTICK_PERIOD_MS);
256
257#if HARDWARE == HW_PCB_REV1
258 gpio_config_t io_conf = {
259 .pin_bit_mask = 1ULL << GPIO_CODEC_NRESET,
260 .mode = GPIO_MODE_OUTPUT,
261 .pull_up_en = GPIO_PULLUP_DISABLE,
262 .pull_down_en = GPIO_PULLDOWN_DISABLE,
263 .intr_type = GPIO_INTR_DISABLE,
264 };
265
266 esp_err_t rc = gpio_set_level(GPIO_CODEC_NRESET, 0);
267 if(rc != ESP_OK)
268 {
269 ESP_LOGE(TAG, "gpio_set_level failed %d, %s", rc, esp_err_to_name(rc));
270 }
271
272 rc = gpio_config(&io_conf);
273 if(rc != ESP_OK)
274 {
275 ESP_LOGE(TAG, "gpio_config failed %d, %s", rc, esp_err_to_name(rc));
276 }
277
278 vTaskDelay(5 / portTICK_PERIOD_MS);
279
280 rc = gpio_set_level(GPIO_CODEC_NRESET, 1);
281 if(rc != ESP_OK)
282 {
283 ESP_LOGE(TAG, "gpio_set_level failed %d, %s", rc, esp_err_to_name(rc));
284 }
285
286 vTaskDelay(5 / portTICK_PERIOD_MS);
287
288#endif
289
290 pcm3060_init(I2C_NUM, 0);
291 rc = pcm3060_power_up();
292 if(rc != ESP_OK)
293 {
294 ESP_LOGE(TAG, "pcm3060 power up failed");
295 }
296
297 /* Start the mdns service */
298 start_mdns();
299
300#if 0
301 rc = xTaskCreate(profiling_task,
302 "i2s profiling",
303 2000,
304 NULL,
305 tskIDLE_PRIORITY + 2,
306 NULL);
307 if(rc != pdPASS)
308 {
309 ESP_LOGE(TAG, "Profiling task create failed %d", rc);
310 }
311#endif
312
313 esp_netif_create_default_wifi_sta();
314 esp_netif_create_default_wifi_ap();
315 wifi_init_config_t cfg = WIFI_INIT_CONFIG_DEFAULT();
316 ESP_ERROR_CHECK(esp_wifi_init(&cfg));
317 ESP_ERROR_CHECK(esp_wifi_set_ps(WIFI_PS_NONE));
318
319 auto wifi_queue = WifiQueue();
320
321 auto wifi_send_event = [&](wifi::InternalEvent event)
322 {
323 auto rc = wifi_queue.send(&event, 0);
324 if(rc != pdPASS)
325 {
326 ESP_LOGE(TAG, "Failed to post wifi event to queue");
327 }
328 };
329
330 auto in_queue = new Queue<AppMessage, QUEUE_LEN>;
331 auto out_queue = new Queue<AppMessage, QUEUE_LEN>;
332 auto server = new Server(in_queue, out_queue);
333
334 auto app_send_event = [&](wifi::UserEvent event)
335 {
336 switch(event)
337 {
338 case wifi::UserEvent::CONNECTED:
339 ESP_LOGI(TAG, "Starting webserver");
340 server->start();
341 break;
342 case wifi::UserEvent::DISCONNECTED:
343 ESP_LOGI(TAG, "Stopping webserver");
344 server->stop();
345 break;
346 default:
347 ESP_LOGE(TAG, "Unhandled event %d", event.get_value());
348 break;
349 }
350 };
351
352 static auto wifi = wifi::WifiStateMachine(wifi_send_event, app_send_event);
353
354 auto wifi_state_chart =
355 etl::state_chart_ct<wifi::WifiStateMachine,
356 wifi,
357 wifi::WifiStateMachine::transition_table,
358 ARRAY_LENGTH(
359 wifi::WifiStateMachine::transition_table),
360 wifi::WifiStateMachine::state_table,
361 ARRAY_LENGTH(wifi::WifiStateMachine::state_table),
362 wifi::State::INIT>();
363 wifi_state_chart.start();
364
365 /* Register event handlers to send events on to wifi state machine */
366 ESP_ERROR_CHECK(esp_event_handler_register(
367 IP_EVENT, IP_EVENT_STA_GOT_IP, connect_handler, &wifi_queue));
368 ESP_ERROR_CHECK(esp_event_handler_register(WIFI_EVENT,
369 WIFI_EVENT_STA_DISCONNECTED,
370 disconnect_handler,
371 &wifi_queue));
372 ESP_ERROR_CHECK(esp_event_handler_register(
373 WIFI_EVENT, WIFI_EVENT_STA_START, wifi_start_handler, &wifi_queue));
374
375 auto midi_uart = new midi::EspMidiUart(UART_NUM_MIDI, GPIO_NUM_MIDI);
376
377 debug_dump_task_list();
378
379 auto send_message = [&](const AppMessage &message)
380 { server->send_message(message); };
381
382 auto main_thread = MainThread<MAX_PARAMETERS, QUEUE_LEN>(
383 send_message,
384 *in_queue,
385 midi_uart,
386 audio_params,
387 persistence,
388 std::make_unique<Crud>("nvs", "midi_mapping"),
389 std::make_unique<Crud>("nvs", "presets"));
390
391 audio::i2s_setup(PROFILING_GPIO, audio_params.get());
392
393 ESP_LOGI(TAG, "setup done");
394 ESP_LOGI(TAG, "stack: %d", uxTaskGetStackHighWaterMark(NULL));
395 heap_caps_print_heap_info(MALLOC_CAP_DEFAULT);
396
397 while(1)
398 {
399 vTaskDelay(pdMS_TO_TICKS(10));
400 auto tick_count_start = xTaskGetTickCount();
401
402 main_thread.loop();
403
404 {
405 // i2s produces an event on each buffer TX/RX. We process all the
406 // events in the current iteration, so that the queue doesn't fill
407 // up.
408 i2s_event_t event;
409 while(xQueueReceive(audio::i2s_queue, &event, 0))
410 {
411 audio::log_i2s_event(event);
412 }
413 }
414
415 wifi::InternalEvent wifi_event;
416 while(pdPASS == wifi_queue.receive(&wifi_event, 0))
417 {
418 wifi::State state{wifi_state_chart.get_state_id()};
419 ESP_LOGI(
420 TAG, "state: %s event: %s", state.c_str(), wifi_event.c_str());
421
422 wifi_state_chart.process_event(wifi_event);
423
424 wifi::State new_state{wifi_state_chart.get_state_id()};
425 if(new_state != state)
426 {
427 ESP_LOGI(TAG, "changed to state: %s", new_state.c_str());
428 }
429 }
430
431 /* Check if the current iteration of the main loop took too long. This
432 * might be caused by blocking while handling events.
433 *
434 * It can also be caused by another thread that has a higher priority
435 * than the main thread.
436 */
437 auto tick_count_iteration = xTaskGetTickCount() - tick_count_start;
438 if(tick_count_iteration > pdMS_TO_TICKS(5))
439 {
440 ESP_LOGW(TAG,
441 "slow iteration: %d ms",
442 (int)pdTICKS_TO_MS(tick_count_iteration));
443 }
444 }
445}

447void nvs_debug_print()
448{
449 ESP_LOGI(TAG, "dumping NVS using C interface");
450 nvs_iterator_t it = NULL;
451 esp_err_t res = nvs_entry_find("nvs", "persistence", NVS_TYPE_ANY, &it);
452 while(res == ESP_OK)
453 {
454 nvs_entry_info_t info;
455 nvs_entry_info(
456 it,
457 &info); // Can omit error check if parameters are guaranteed to be non-NULL
458 ESP_LOGI(TAG, "key '%s', type '%d'", info.key, info.type);
459 res = nvs_entry_next(&it);
460 }
461 nvs_release_iterator(it);
462}
