31template <typename AudioParametersT,
32 typename MappingManagerT,
33 typename PresetLoaderT>
34class MidiMessageHandler
35{
36public:
37 MidiMessageHandler(std::shared_ptr<AudioParametersT> a_parameters,
38 std::shared_ptr<MappingManagerT> a_mapping_manager,
39 std::shared_ptr<PresetLoaderT> a_preset_loader)
40 : parameters{std::move(a_parameters)},
41 mapping_manager{std::move(a_mapping_manager)},
42 preset_loader{std::move(a_preset_loader)} {};
43
47 void process_message(midi::Message message) const
48 {
49 auto cc_params =
50 get_if<midi::Message::ControlChange>(&message.parameters);
51 if(!cc_params)
52 return;
53
54 for(const auto &[_, mapping] : *mapping_manager)
55 {
56 if(mapping.midi_channel != message.channel)
57 {
58 continue;
59 }
60
61 if(mapping.cc_number != cc_params->control)
62 {
63 continue;
64 }
65
66 switch(mapping.mode)
67 {
68 case midi::Mapping::Mode::PARAMETER:
69 parameters->update(mapping.parameter_name,
70 cc_params->value /
71 float(midi::CC_VALUE_MAX));
72 break;
73 case midi::Mapping::Mode::TOGGLE:
74 {
75 if(cc_params->value == 0)
76 {
77 continue;
78 }
79
80 auto old_value = parameters->get(mapping.parameter_name);
81
82 parameters->update(mapping.parameter_name,
83 old_value < 0.5f ? 1.f : 0.f);
84 break;
85 }
86 case midi::Mapping::Mode::BUTTON:
87 {
88 // A button sends the maximum value when it becomes pressed.
89 if(cc_params->value != midi::CC_VALUE_MAX)
90 {
91 continue;
92 }
93
94 auto id = mapping.preset_id;
95 preset_loader->load_preset(id);
96 break;
97 }
98 }
99 }
100 }
101
102private:
103 std::shared_ptr<AudioParametersT> parameters;
104 std::shared_ptr<MappingManagerT> mapping_manager;
105 std::shared_ptr<PresetLoaderT> preset_loader;
106};
