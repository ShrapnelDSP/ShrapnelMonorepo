30template <>
31std::optional<std::span<uint8_t>>
32to_bytes(const shrapnel_messages_Message &message, std::span<uint8_t> buffer)
33{
34 pb_ostream_t stream = pb_ostream_from_buffer(buffer.data(), buffer.size());
35
36 bool success = pb_encode(&stream, &shrapnel_messages_Message_msg, &message);
37 if(!success)
38 {
39 return std::nullopt;
40 }
41
42 return buffer.subspan(0, stream.bytes_written);
43}

61template <>
62int to_proto(const ApiMessage &message, shrapnel_messages_Message &out)
63{
64 return std::visit(
65 [&out](const auto &message) -> int
66 {
67 using T = std::decay_t<decltype(message)>;
68 if constexpr(std::is_same_v<T, events::ApiMessage>)
69 {
70 out.which_message = shrapnel_messages_Message_audio_event_tag;
71 return to_proto<shrapnel_audio_events_Message>(
72 message, out.message.audio_event);
73 }
74 else if constexpr(std::is_same_v<T, parameters::ApiMessage>)
75 {
76 out.which_message = shrapnel_messages_Message_command_tag;
77 return to_proto<shrapnel_cmd_handling_Message>(
78 message, out.message.command);
79 }
80 else if constexpr(std::is_same_v<T, midi::MappingApiMessage>)
81 {
82 out.which_message = shrapnel_messages_Message_midi_mapping_tag;
83 return to_proto<shrapnel_midi_mapping_Message>(
84 message, out.message.midi_mapping);
85 }
86 else if constexpr(std::is_same_v<T, presets::PresetsApiMessage>)
87 {
88 out.which_message = shrapnel_messages_Message_preset_tag;
89 return to_proto<shrapnel_presets_Message>(message,
90 out.message.preset);
91 }
92 else if constexpr(std::is_same_v<
93 T,
94 selected_preset::SelectedPresetApiMessage>)
95 {
96 out.which_message =
97 shrapnel_messages_Message_selected_preset_tag;
98 return to_proto<shrapnel_selected_preset_Message>(
99 message, out.message.selected_preset);
100 }
101 else
102 {
103 return -1;
104 }
105 },
106 message);
107}
