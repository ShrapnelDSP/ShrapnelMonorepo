32static uint32_t start_cycles = 0;
33static uint32_t stage_cycles[NUMBER_OF_STAGES] = {};
34SemaphoreHandle_t profiling_mutex;
35static bool got_semaphore = false;
36static uint32_t cpu_freq_mhz;

38void profiling_start(void)
39{
40 if(xSemaphoreTake(profiling_mutex, 0))
41 {
42 got_semaphore = true;
43 start_cycles = cpu_hal_get_cycle_count();
44 }
45}

47void profiling_mark_stage(unsigned int stage)
48{
49 assert(stage < NUMBER_OF_STAGES);
50
51 if(got_semaphore)
52 {
53 stage_cycles[stage] = cpu_hal_get_cycle_count();
54 }
55}

57void profiling_stop(void)
58{
59 if(got_semaphore)

61 xSemaphoreGive(profiling_mutex);
62 got_semaphore = false;

66static double cycles_to_percent(int64_t cycles)
67{
68 float cycles_per_second = cpu_freq_mhz * 1e6f;
69
70 float ratio = cycles /
71 ((float)DMA_BUF_SIZE / SAMPLE_RATE) /
72 (cycles_per_second);
73
74 return (double)(100 * ratio);
75}

77static int64_t cycles_to_us(int64_t cycles)
78{
79 return cycles / cpu_freq_mhz;
80}

82void i2s_profiling_task(void *param)
83{
84 (void) param;
85
86 cpu_freq_mhz = ets_get_cpu_frequency();
87 ESP_LOGI(TAG, "CPU frequency: %d MHz", cpu_freq_mhz);
88
89 while(1)
90 {
91 vTaskDelay(1000 / portTICK_PERIOD_MS);
92
93 if(xSemaphoreTake(profiling_mutex, 100 / portTICK_PERIOD_MS))
94 {
95 ESP_LOGI(TAG, " ========= \n");
96
97 int64_t total_cycles = 0;
98 for(int i = 0; i < NUMBER_OF_STAGES; i++)
99 {
100 //if the stage has not been assigned, stop printing
101 if(stage_cycles[i] == 0)
102 {
103 break;
104 }
105
106 int64_t current_stage_cycles = stage_cycles[i] - ((i == 0) ? start_cycles : stage_cycles[i - 1]);
107 ESP_LOGI(TAG, "Stage %3d processing took %7lld cycles %4lld us (%03.1f %%)",
108 i,
109 current_stage_cycles,
110 cycles_to_us(current_stage_cycles),
111 cycles_to_percent(current_stage_cycles));
112
113 total_cycles += current_stage_cycles;
114 }
115
116 ESP_LOGI(TAG, "Total processing took %7lld cycles %4lld us (%03.1f %%)",
117 total_cycles,
118 cycles_to_us(total_cycles),
119 cycles_to_percent(total_cycles));
120
121 ESP_LOGI(TAG, " ========= \n");
122
123 xSemaphoreGive(profiling_mutex);
124 }
125 else
126 {
127 ESP_LOGE(TAG, "%s failed to take semaphore", __FUNCTION__);
128 }
129 }
130}
