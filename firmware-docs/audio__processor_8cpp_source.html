55void AudioProcessor::process(std::span<float, 512> fbuf)
56{
57 gate_set_threshold(*parameters.gate.threshold, *parameters.gate.hysteresis);
58 gate_set_attack(*parameters.gate.attack);
59 gate_set_hold(*parameters.gate.hold);
60 gate_set_release(*parameters.gate.release);
61
62 profiling_mark_stage(0);
63
64 gate_analyse(fbuf.data(), fbuf.size());
65 profiling_mark_stage(1);
66
67 valvestate.set_gain(*parameters.amplifier.gain,
68 *parameters.amplifier.channel);
69 valvestate.set_fmv(*parameters.amplifier.bass,
70 *parameters.amplifier.middle,
71 *parameters.amplifier.treble);
72 valvestate.set_contour(*parameters.amplifier.contour);
73 valvestate.set_volume(decibel_to_ratio(*parameters.amplifier.volume));
74 profiling_mark_stage(2);
75
76 valvestate.process(fbuf);
77 profiling_mark_stage(3);
78
79 if(*parameters.gate.bypass < 0.5f)
80 {
81 gate_process(fbuf.data(), fbuf.size());
82 }
83 profiling_mark_stage(4);
84
85 /* speaker IR */
86 speaker.process(fbuf);
87 profiling_mark_stage(17);
88
89 chorus.set_modulation_rate_hz(*parameters.chorus.rate);
90 chorus.set_modulation_depth(*parameters.chorus.depth);
91 chorus.set_modulation_mix(*parameters.chorus.mix);
92 profiling_mark_stage(18);
93
94 if(*parameters.chorus.bypass < 0.5f)
95 {
96 chorus.process(fbuf);
97 }
98 profiling_mark_stage(19);
99
100 wah.set_expression_position(*parameters.wah.position);
101 wah.set_vocal(*parameters.wah.vocal);
102
103 if(*parameters.wah.bypass < 0.5f)
104 {
105 wah.process(fbuf);
106 }
107}
